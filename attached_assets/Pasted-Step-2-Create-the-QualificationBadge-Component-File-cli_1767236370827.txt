// QUALIFICATION BADGE COMPONENT
// Visual indicator for driver/trailer qualification status
// ============================================

import React from 'react';
import { AlertTriangle, CheckCircle, AlertCircle, Info } from 'lucide-react';
import {
  Tooltip,
  TooltipContent,
  TooltipProvider,
  TooltipTrigger,
} from '@/components/ui/tooltip';

// ============================================
// TYPES
// ============================================

interface QualificationBadgeProps {
  isQualified: boolean;
  issueCount: number;
  warningCount: number;
  primaryIssue?: string | null;
  warnings?: string[];
  size?: 'sm' | 'md' | 'lg';
  showLabel?: boolean;
}

// ============================================
// COMPONENT
// ============================================

export function QualificationBadge({ 
  isQualified, 
  issueCount, 
  warningCount, 
  primaryIssue,
  warnings = [],
  size = 'md',
  showLabel = false
}: QualificationBadgeProps) {
  
  // Size classes for the icon
  const sizeClasses = {
    sm: 'h-4 w-4',
    md: 'h-5 w-5',
    lg: 'h-6 w-6'
  };

  const iconSize = sizeClasses[size];

  // ============================================
  // STATE 1: Fully qualified, no warnings
  // ============================================
  if (isQualified && warningCount === 0) {
    return (
      <TooltipProvider>
        <Tooltip>
          <TooltipTrigger asChild>
            <div className="flex items-center gap-1.5 cursor-help">
              <CheckCircle className={`${iconSize} text-green-500`} />
              {showLabel && (
                <span className="text-xs text-green-500 font-medium">Qualified</span>
              )}
            </div>
          </TooltipTrigger>
          <TooltipContent side="top" className="bg-gray-800 border-gray-700">
            <p className="text-green-400 font-medium">✓ Fully qualified to tow</p>
          </TooltipContent>
        </Tooltip>
      </TooltipProvider>
    );
  }

  // ============================================
  // STATE 2: Qualified but has warnings
  // ============================================
  if (isQualified && warningCount > 0) {
    return (
      <TooltipProvider>
        <Tooltip>
          <TooltipTrigger asChild>
            <div className="flex items-center gap-1.5 cursor-help">
              <Info className={`${iconSize} text-yellow-500`} />
              {showLabel && (
                <span className="text-xs text-yellow-500 font-medium">
                  {warningCount} warning{warningCount > 1 ? 's' : ''}
                </span>
              )}
            </div>
          </TooltipTrigger>
          <TooltipContent side="top" className="bg-gray-800 border-gray-700 max-w-xs">
            <p className="text-yellow-400 font-medium mb-1">
              ⚠ Qualified with {warningCount} warning{warningCount > 1 ? 's' : ''}
            </p>
            {warnings.length > 0 ? (
              <ul className="text-xs text-gray-300 space-y-1">
                {warnings.slice(0, 3).map((warning, i) => (
                  <li key={i} className="flex items-start gap-1">
                    <span className="text-yellow-500 mt-0.5">•</span>
                    <span>{warning}</span>
                  </li>
                ))}
                {warnings.length > 3 && (
                  <li className="text-gray-500">+ {warnings.length - 3} more</li>
                )}
              </ul>
            ) : primaryIssue && (
              <p className="text-xs text-gray-300">{primaryIssue}</p>
            )}
          </TooltipContent>
        </Tooltip>
      </TooltipProvider>
    );
  }

  // ============================================
  // STATE 3: Not qualified (has blocking issues)
  // ============================================
  return (
    <TooltipProvider>
      <Tooltip>
        <TooltipTrigger asChild>
          <div className="flex items-center gap-1.5 cursor-help">
            <AlertTriangle className={`${iconSize} text-red-500`} />
            {showLabel && (
              <span className="text-xs text-red-500 font-medium">
                Not Qualified
              </span>
            )}
          </div>
        </TooltipTrigger>
        <TooltipContent side="top" className="bg-gray-800 border-gray-700 max-w-xs">
          <p className="text-red-400 font-medium mb-1">
            ✗ Not qualified to tow
          </p>
          {primaryIssue && (
            <p className="text-xs text-gray-300 mb-1">{primaryIssue}</p>
          )}
          {issueCount > 1 && (
            <p className="text-xs text-gray-500">
              + {issueCount - 1} more issue{issueCount > 2 ? 's' : ''}
            </p>
          )}
        </TooltipContent>
      </Tooltip>
    </TooltipProvider>
  );
}

// ============================================
// COMPACT VARIANT - Just the icon, minimal space
// ============================================

export function QualificationBadgeCompact({ 
  isQualified, 
  warningCount = 0 
}: { 
  isQualified: boolean; 
  warningCount?: number;
}) {
  if (isQualified && warningCount === 0) {
    return <CheckCircle className="h-4 w-4 text-green-500" />;
  }
  if (isQualified && warningCount > 0) {
    return <Info className="h-4 w-4 text-yellow-500" />;
  }
  return <AlertTriangle className="h-4 w-4 text-red-500" />;
}

// ============================================
// INLINE STATUS - For use in text/lists
// ============================================

export function QualificationStatus({ 
  isQualified, 
  issueCount = 0,
  warningCount = 0 
}: { 
  isQualified: boolean; 
  issueCount?: number;
  warningCount?: number;
}) {
  if (isQualified && warningCount === 0) {
    return (
      <span className="inline-flex items-center gap-1 text-green-500 text-sm">
        <CheckCircle className="h-3.5 w-3.5" />
        Qualified
      </span>
    );
  }
  if (isQualified && warningCount > 0) {
    return (
      <span className="inline-flex items-center gap-1 text-yellow-500 text-sm">
        <Info className="h-3.5 w-3.5" />
        {warningCount} warning{warningCount > 1 ? 's' : ''}
      </span>
    );
  }
  return (
    <span className="inline-flex items-center gap-1 text-red-500 text-sm">
      <AlertTriangle className="h-3.5 w-3.5" />
      {issueCount} issue{issueCount > 1 ? 's' : ''}
    </span>
  );
}
