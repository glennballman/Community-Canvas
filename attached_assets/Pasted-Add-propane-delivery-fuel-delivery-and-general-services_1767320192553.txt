-- ============================================================================
-- SUMMARY OF ALL PROVIDERS
-- ============================================================================

SELECT 
    ssp.provider_type,
    COUNT(*) as count,
    ROUND(AVG(ssp.overall_rating), 2) as avg_rating,
    SUM(ssp.review_count) as total_reviews
FROM staging_service_providers ssp
GROUP BY ssp.provider_type
ORDER BY count DESC;
