import React, { useState, useEffect, useCallback } from 'react';

interface Tenant {
    id: string;
    name: string;
    slug: string;
    tenant_type: string;
    business_type: string;
    government_level: string;
    city: string;
    province: string;
    status: string;
    created_at: string;
    member_count: number;
    owner_email: string;
    owner_first_name: string;
    owner_last_name: string;
}

interface TenantMember {
    id: string;
    email: string;
    first_name: string;
    last_name: string;
    avatar_url: string;
    role: string;
    title: string;
    status: string;
    joined_at: string;
}

export default function TenantsManagement() {
    const [tenants, setTenants] = useState<Tenant[]>([]);
    const [selectedTenant, setSelectedTenant] = useState<Tenant | null>(null);
    const [tenantMembers, setTenantMembers] = useState<TenantMember[]>([]);
    const [loading, setLoading] = useState(true);
    const [detailLoading, setDetailLoading] = useState(false);

    // Filters
    const [typeFilter, setTypeFilter] = useState('');
    const [search, setSearch] = useState('');

    const token = localStorage.getItem('cc_token');

    const loadTenants = useCallback(async () => {
        setLoading(true);
        try {
            let url = '/api/foundation/tenants?';
            if (typeFilter) url += `type=${typeFilter}&`;
            if (search) url += `search=${encodeURIComponent(search)}`;

            const res = await fetch(url, {
                headers: { Authorization: `Bearer ${token}` }
            });
            const data = await res.json();

            if (data.success) {
                setTenants(data.tenants);
            }
        } catch (err) {
            console.error('Failed to load tenants:', err);
        } finally {
            setLoading(false);
        }
    }, [typeFilter, search, token]);

    useEffect(() => {
        loadTenants();
    }, [loadTenants]);

    async function loadTenantDetails(tenantId: string) {
        setDetailLoading(true);
        try {
            const res = await fetch(`/api/foundation/tenants/${tenantId}`, {
                headers: { Authorization: `Bearer ${token}` }
            });
            const data = await res.json();

            if (data.success) {
                setSelectedTenant(data.tenant);
                setTenantMembers(data.members || []);
            }
        } catch (err) {
            console.error('Failed to load tenant details:', err);
        } finally {
            setDetailLoading(false);
        }
    }

    const typeColors: Record<string, string> = {
        platform: 'bg-purple-500 text-white',
        government: 'bg-blue-500 text-white',
        business: 'bg-green-500 text-white',
        property: 'bg-yellow-500 text-black',
        individual: 'bg-gray-500 text-white',
    };

    const typeIcons: Record<string, string> = {
        platform: 'üèõÔ∏è',
        government: 'üè¢',
        business: 'üíº',
        property: 'üè†',
        individual: 'üë§',
    };

    // Group tenants by type
    const groupedTenants = tenants.reduce((acc, t) => {
        if (!acc[t.tenant_type]) acc[t.tenant_type] = [];
        acc[t.tenant_type].push(t);
        return acc;
    }, {} as Record<string, Tenant[]>);

    const typeOrder = ['platform', 'government', 'business', 'property', 'individual'];

    return (
        <div className="p-6">
            <div className="flex items-center justify-between mb-6">
                <div>
                    <h1 className="text-2xl font-bold text-white">Tenant Management</h1>
                    <p className="text-gray-400">Manage organizations and their members</p>
                </div>
                <button className="bg-blue-600 hover:bg-blue-700 text-white px-4 py-2 rounded-lg">
                    + Create Tenant
                </button>
            </div>

            {/* Stats */}
            <div className="grid grid-cols-5 gap-4 mb-6">
                {typeOrder.map(type => (
                    <div 
                        key={type}
                        onClick={() => setTypeFilter(typeFilter === type ? '' : type)}
                        className={`bg-gray-800 rounded-lg p-4 cursor-pointer transition-colors ${
                            typeFilter === type ? 'ring-2 ring-blue-500' : 'hover:bg-gray-750'
                        }`}
                    >
                        <div className="text-2xl mb-1">{typeIcons[type]}</div>
                        <div className="text-2xl font-bold text-white">
                            {tenants.filter(t => t.tenant_type === type).length}
                        </div>
                        <div className="text-sm text-gray-400 capitalize">{type}</div>
                    </div>
                ))}
            </div>

            {/* Search */}
            <div className="bg-gray-800 rounded-lg p-4 mb-6">
                <input
                    type="text"
                    placeholder="Search tenants by name..."
                    value={search}
                    onChange={(e) => setSearch(e.target.value)}
                    className="w-full bg-gray-700 border border-gray-600 rounded-lg px-4 py-2 text-white placeholder-gray-400"
                />
            </div>

            <div className="grid grid-cols-1 lg:grid-cols-3 gap-6">
                {/* Tenant List */}
                <div className="lg:col-span-2 space-y-6">
                    {loading ? (
                        <div className="bg-gray-800 rounded-lg p-8 text-center text-gray-500">
                            Loading tenants...
                        </div>
                    ) : (
                        typeOrder.filter(type => !typeFilter || type === typeFilter).map(type => {
                            const typeTenants = groupedTenants[type] || [];
                            if (typeTenants.length === 0) return null;

                            return (
                                <div key={type} className="bg-gray-800 rounded-lg overflow-hidden">
                                    <div className={`px-4 py-2 ${typeColors[type]} flex items-center gap-2`}>
                                        <span>{typeIcons[type]}</span>
                                        <span className="font-medium capitalize">{type} Tenants</span>
                                        <span className="ml-auto text-sm opacity-75">{typeTenants.length}</span>
                                    </div>
                                    <div className="divide-y divide-gray-700">
                                        {typeTenants.map(tenant => (
                                            <div
                                                key={tenant.id}
                                                onClick={() => loadTenantDetails(tenant.id)}
                                                className={`px-4 py-3 cursor-pointer hover:bg-gray-750 ${
                                                    selectedTenant?.id === tenant.id ? 'bg-gray-700' : ''
                                                }`}
                                            >
                                                <div className="flex items-center justify-between">
                                                    <div>
                                                        <div className="text-white font-medium">{tenant.name}</div>
                                                        <div className="text-gray-400 text-sm">
                                                            {tenant.city && `${tenant.city}, `}
                                                            {tenant.slug}
                                                        </div>
                                                    </div>
                                                    <div className="text-right">
                                                        <div className="text-blue-400">{tenant.member_count || 0} members</div>
                                                        <div className="text-gray-500 text-xs">
                                                            {tenant.status}
                                                        </div>
                                                    </div>
                                                </div>
                                            </div>
                                        ))}
                                    </div>
                                </div>
                            );
                        })
                    )}
                </div>

                {/* Tenant Detail Panel */}
                <div className="lg:col-span-1">
                    {selectedTenant ? (
                        <div className="bg-gray-800 rounded-lg p-4 space-y-4 sticky top-4">
                            {detailLoading ? (
                                <div className="text-center py-8 text-gray-500">Loading...</div>
                            ) : (
                                <>
                                    {/* Header */}
                                    <div className="pb-4 border-b border-gray-700">
                                        <div className={`inline-block px-3 py-1 rounded text-sm mb-2 ${typeColors[selectedTenant.tenant_type]}`}>
                                            {typeIcons[selectedTenant.tenant_type]} {selectedTenant.tenant_type}
                                        </div>
                                        <h3 className="text-xl font-bold text-white">{selectedTenant.name}</h3>
                                        <p className="text-gray-400">/{selectedTenant.slug}</p>
                                    </div>

                                    {/* Details */}
                                    <div className="space-y-2 text-sm">
                                        {selectedTenant.city && (
                                            <div className="flex justify-between">
                                                <span className="text-gray-400">Location</span>
                                                <span className="text-white">{selectedTenant.city}, {selectedTenant.province}</span>
                                            </div>
                                        )}
                                        {selectedTenant.business_type && (
                                            <div className="flex justify-between">
                                                <span className="text-gray-400">Business Type</span>
                                                <span className="text-white">{selectedTenant.business_type}</span>
                                            </div>
                                        )}
                                        {selectedTenant.government_level && (
                                            <div className="flex justify-between">
                                                <span className="text-gray-400">Level</span>
                                                <span className="text-white">{selectedTenant.government_level}</span>
                                            </div>
                                        )}
                                        <div className="flex justify-between">
                                            <span className="text-gray-400">Status</span>
                                            <span className="text-green-400">{selectedTenant.status}</span>
                                        </div>
                                        <div className="flex justify-between">

                                    {/* Members */}
                                    <div>

                                            <button className="text-blue-400 text-xs hover:underline">+ Add</button>
                                        </div>
                                        <div className="space-y-2 max-h-60 overflow-y-auto">
                                            {tenantMembers.map(m => (
                                                <div key={m.id} className="bg-gray-700 rounded p-2 flex items-center gap-2">
                                                    <div className="w-8 h-8 bg-blue-600 rounded-full flex items-center justify-center text-white text-sm">
                                                        {(m.first_name?.[0] || m.email[0]).toUpperCase()}
                                                    </div>
                                                    <div className="flex-1 min-w-0">
                                                        <div className="text-white text-sm truncate">
                                                            {m.first_name} {m.last_name}

                                    </div>

                                            Edit Tenant

                                        <button className="w-full bg-gray-700 hover:bg-gray-600 text-white py-2 rounded text-sm">

                                        </button>
                                    </div>
                                </>
                            )}
                        </div>
                    ) : (
                        <div className="bg-gray-800 rounded-lg p-8 text-center text-gray-500">

                            <p>Select a tenant to view details</p>
                        </div>
                    )}
                </div>
            </div>
        </div>
    );
}
