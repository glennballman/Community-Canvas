-- Summary of truck stops
SELECT 
    name, 
    city, 
    num_truck_spots,
    CASE WHEN has_cat_scale THEN 'Yes' ELSE 'No' END as cat_scale,
    CASE WHEN has_showers THEN 'Yes' ELSE 'No' END as showers,
    CASE WHEN has_restaurant THEN 'Yes' ELSE 'No' END as restaurant,
    trucker_score
FROM staging_properties
WHERE property_type = 'parking_lot' AND property_subtype = 'truck_stop'
ORDER BY trucker_score DESC;
