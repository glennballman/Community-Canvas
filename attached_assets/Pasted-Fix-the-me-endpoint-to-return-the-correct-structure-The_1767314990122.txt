// Make sure verifySession uses this formatter:
async verifySession(token: string): Promise<{ valid: boolean; host?: any }> {
    const result = await db.query(`
        SELECT s.*, h.*
        FROM host_sessions s
        JOIN host_accounts h ON h.id = s.host_account_id
        WHERE s.session_token = $1 
          AND s.is_valid = true 
          AND s.expires_at > NOW()
    `, [token]);

    if (result.rows.length === 0) {
        return { valid: false };
    }

    const row = result.rows[0];

    // Update last used
    await db.query(
        'UPDATE host_sessions SET last_used_at = NOW() WHERE session_token = $1',
        [token]
    );

    return { 
        valid: true, 
        host: this.formatHost(row)  // Use the formatter!
    };
}
