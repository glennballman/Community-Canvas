Create the Trip Planning UI with participant profiles, vehicle assessments, and trip qualification checks. This will allow users to:
1. Create their profile with skills and certifications
2. Add and assess their vehicle
3. Check if they're qualified for a specific trip
4. See what gaps they need to fill (training, equipment, vehicle upgrades)

import React, { useState } from 'react';
import { ParticipantProfile, ParticipantSkill, SkillCategory, SkillLevel, skillCategoryIcons, skillLevelColors } from '../../types/tripPlanning';

interface ParticipantProfileFormProps {
  participant: ParticipantProfile | null;
  onSave: (participant: ParticipantProfile) => void;
  onCancel: () => void;
}

const SKILL_TYPES: Record<SkillCategory, string[]> = {
  paddling: ['sea_kayak', 'whitewater', 'canoe', 'sup', 'self_rescue'],
  driving: ['standard', 'mountain', 'gravel_road', 'winter', 'commercial', 'towing'],
  backcountry: ['navigation', 'camping', 'wilderness_first_aid', 'bear_aware', 'leave_no_trace'],
  water_safety: ['vhf_radio', 'boating_license', 'fishing_license_salt', 'fishing_license_fresh'],
  emergency: ['cpr_first_aid', 'wilderness_first_aid', 'satellite_device', 'emergency_beacon']
};

const SKILL_TYPE_LABELS: Record<string, string> = {
  sea_kayak: 'Sea Kayaking',
  whitewater: 'Whitewater',
  canoe: 'Canoeing',
  sup: 'Stand-Up Paddleboard',
  self_rescue: 'Self-Rescue',
  standard: 'Standard Driving',
  mountain: 'Mountain Driving',
  gravel_road: 'Gravel Road',
  winter: 'Winter Driving',
  commercial: 'Commercial License',
  towing: 'Towing',
  navigation: 'Navigation (Map/Compass)',
  camping: 'Backcountry Camping',
  wilderness_first_aid: 'Wilderness First Aid',
  bear_aware: 'Bear Awareness',
  leave_no_trace: 'Leave No Trace',
  vhf_radio: 'VHF Radio',
  boating_license: 'Boating License',
  fishing_license_salt: 'Fishing License (Saltwater)',
  fishing_license_fresh: 'Fishing License (Freshwater)',
  cpr_first_aid: 'CPR/First Aid',
  satellite_device: 'Satellite Communicator',
  emergency_beacon: 'Emergency Beacon (PLB)'
};

export function ParticipantProfileForm({ participant, onSave, onCancel }: ParticipantProfileFormProps) {
  const [formData, setFormData] = useState<Partial<ParticipantProfile>>({
    name: participant?.name || '',
    email: participant?.email || '',
    phone: participant?.phone || '',
    emergency_contact_name: participant?.emergency_contact_name || '',
    emergency_contact_phone: participant?.emergency_contact_phone || '',
    country_of_origin: participant?.country_of_origin || '',
    languages: participant?.languages || ['English'],
    medical_conditions: participant?.medical_conditions || [],
    dietary_restrictions: participant?.dietary_restrictions || [],
    fitness_level: participant?.fitness_level || 5,
    swimming_ability: participant?.swimming_ability || 'basic'
  });

  const [skills, setSkills] = useState<Partial<ParticipantSkill>[]>(
    participant?.skills || []
  );

  const [activeSection, setActiveSection] = useState<'basic' | 'skills' | 'emergency'>('basic');
  const [saving, setSaving] = useState(false);

  const [newSkill, setNewSkill] = useState<Partial<ParticipantSkill>>({
    skill_category: 'paddling',
    skill_type: 'sea_kayak',
    skill_level: 'beginner'
  });

  const handleSave = async () => {
    if (!formData.name) {
      alert('Please enter your name');
      return;
    }

    setSaving(true);
    try {
      // Save participant
      const response = await fetch('/api/v1/planning/participants', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(formData)
      });
      
      if (!response.ok) throw new Error('Failed to save participant');
      
      const savedParticipant = await response.json();

      // Save skills
      for (const skill of skills) {
        if (skill.skill_category && skill.skill_type && skill.skill_level) {
          await fetch(`/api/v1/planning/participants/${savedParticipant.id}/skills`, {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify(skill)
          });
        }
      }

      // Fetch complete profile
      const profileResponse = await fetch(`/api/v1/planning/participants/${savedParticipant.id}`);
      const completeProfile = await profileResponse.json();

      onSave(completeProfile);
    } catch (error) {
      console.error('Error saving participant:', error);
      alert('Failed to save profile. Please try again.');
    } finally {
      setSaving(false);
    }
  };

  const addSkill = () => {
    if (newSkill.skill_category && newSkill.skill_type && newSkill.skill_level) {
      // Check for duplicate
      const exists = skills.some(
        s => s.skill_category === newSkill.skill_category && s.skill_type === newSkill.skill_type
      );
      if (!exists) {
        setSkills([...skills, { ...newSkill }]);
      }
      setNewSkill({
        skill_category: newSkill.skill_category,
        skill_type: SKILL_TYPES[newSkill.skill_category as SkillCategory][0],
        skill_level: 'beginner'
      });
    }
  };

  const removeSkill = (index: number) => {
    setSkills(skills.filter((_, i) => i !== index));
  };

  return (
    <div className="space-y-4">
      {/* Header */}
      <div className="bg-gray-800 rounded-xl p-4">
        <div className="flex items-center justify-between">
          <div className="flex items-center gap-3">
            <button onClick={onCancel} className="text-gray-400 hover:text-white">
              ‚Üê Back
            </button>
            <h2 className="text-xl font-bold text-white">
              {participant ? 'Edit Profile' : 'Create Your Profile'}
            </h2>
          </div>
          <button
            onClick={handleSave}
            disabled={saving}
            className="px-4 py-2 bg-green-600 hover:bg-green-700 disabled:bg-gray-600 text-white rounded-lg font-medium transition"
          >
            {saving ? 'Saving...' : 'Save Profile'}
          </button>
        </div>

        {/* Section Tabs */}
        <div className="flex gap-2 mt-4">
          {(['basic', 'skills', 'emergency'] as const).map(section => (
            <button
              key={section}
              onClick={() => setActiveSection(section)}
              className={`px-4 py-2 rounded-lg font-medium capitalize transition ${
                activeSection === section
                  ? 'bg-blue-600 text-white'
                  : 'bg-gray-700 text-gray-300 hover:bg-gray-600'
              }`}
            >
              {section === 'basic' && 'üë§ '}
              {section === 'skills' && 'üéØ '}
              {section === 'emergency' && 'üö® '}
              {section}
            </button>
          ))}
        </div>
      </div>

      {/* Basic Info Section */}
      {activeSection === 'basic' && (
        <div className="bg-gray-800 rounded-xl p-6 space-y-6">
          <div className="grid grid-cols-1 md:grid-cols-2 gap-6">
            {/* Name */}
            <div>
              <label className="block text-gray-300 text-sm font-medium mb-2">Full Name *</label>
              <input
                type="text"
                value={formData.name}
                onChange={(e) => setFormData({ ...formData, name: e.target.value })}
                className="w-full bg-gray-700 text-white rounded-lg px-4 py-2 border border-gray-600 focus:border-blue-500 focus:outline-none"
                placeholder="Your full name"
              />
            </div>

            {/* Email */}
            <div>
              <label className="block text-gray-300 text-sm font-medium mb-2">Email</label>
              <input
                type="email"
                value={formData.email}
                onChange={(e) => setFormData({ ...formData, email: e.target.value })}
                className="w-full bg-gray-700 text-white rounded-lg px-4 py-2 border border-gray-600 focus:border-blue-500 focus:outline-none"
                placeholder="your@email.com"
              />
            </div>

            {/* Phone */}
            <div>
              <label className="block text-gray-300 text-sm font-medium mb-2">Phone</label>
              <input
                type="tel"

            ‚Üê Back
          </button>
          <div>
            <h2 className="text-xl font-bold text-white">üó∫Ô∏è Route Explorer</h2>
            <p className="text-gray-400 text-sm">View routes, conditions, and alternatives</p>
          </div>
        </div>
      </div>

      {loading ? (
        <div className="bg-gray-800 rounded-xl p-12 text-center">
          <span className="text-4xl animate-spin">‚è≥</span>
          <p className="text-gray-400 mt-2">Loading routes...</p>
        </div>
      ) : (
        <div className="grid grid-cols-1 lg:grid-cols-2 gap-4">
          {/* Route Segments */}
          <div className="bg-gray-800 rounded-xl p-4">
            <h3 className="text-white font-semibold mb-4">Route Segments</h3>
            <div className="space-y-2 max-h-[500px] overflow-y-auto">
              {segments.map(segment => (
                <button
                  key={segment.id}
                  onClick={() => selectSegment(segment)}
                  className={`w-full text-left p-3 rounded-lg transition ${
                    selectedSegment?.id === segment.id
                      ? 'bg-blue-600 text-white'
                      : 'bg-gray-700/50 text-gray-300 hover:bg-gray-700'
                  }`}
                >
                  <div className="flex items-center gap-3">
                    <span className="text-2xl">{routeTypeIcons[segment.route_type] || 'üìç'}</span>
                    <div className="flex-1 min-w-0">
                      <p className="font-medium truncate">{segment.name}</p>
                      <p className="text-sm opacity-80">
                        {segment.distance_km} km ‚Ä¢ {Math.round(segment.typical_duration_minutes / 60 * 10) / 10}h
                      </p>
                    </div>
                    {segment.route_type === 'logging_road' && (
                      <span className="px-2 py-0.5 bg-orange-500/20 text-orange-400 rounded text-xs">
                        Caution
                      </span>
                    )}
                  </div>
                </button>
              ))}
            </div>
          </div>

          {/* Segment Details */}
          <div className="bg-gray-800 rounded-xl p-4">
            {!selectedSegment ? (
              <div className="text-center py-12 text-gray-500">
                <span className="text-4xl">üëà</span>
                <p className="mt-2">Select a route segment to view details</p>
              </div>
            ) : (
              <div className="space-y-4">
                <div>
                  <h3 className="text-white font-semibold text-lg">{selectedSegment.name}</h3>
                  <p className="text-gray-400 text-sm">{selectedSegment.description}</p>
                </div>

                <div className="grid grid-cols-2 gap-3">
                  <div className="bg-gray-700/50 rounded-lg p-3">
                    <p className="text-gray-500 text-xs">From</p>
                    <p className="text-white">{selectedSegment.start_location_name}</p>
                  </div>
                  <div className="bg-gray-700/50 rounded-lg p-3">
                    <p className="text-gray-500 text-xs">To</p>
                    <p className="text-white">{selectedSegment.end_location_name}</p>
                  </div>
                  <div className="bg-gray-700/50 rounded-lg p-3">
                    <p className="text-gray-500 text-xs">Distance</p>
                    <p className="text-white">{selectedSegment.distance_km} km</p>
                  </div>
                  <div className="bg-gray-700/50 rounded-lg p-3">
                    <p className="text-gray-500 text-xs">Duration</p>
                    <p className="text-white">{Math.round(selectedSegment.typical_duration_minutes / 60 * 10) / 10} hours</p>
                  </div>
                </div>

                {/* Requirements */}
                <div>
                  <p className="text-gray-400 text-sm font-medium mb-2">Requirements</p>
                  <div className="flex flex-wrap gap-2">
                    {selectedSegment.winter_tires_required && (
                      <span className="px-2 py-1 bg-blue-500/20 text-blue-400 rounded text-sm">
                        ‚ùÑÔ∏è Winter Tires Required
                      </span>
                    )}
                    {selectedSegment.high_clearance_recommended && (
                      <span className="px-2 py-1 bg-yellow-500/20 text-yellow-400 rounded text-sm">
                        ‚¨ÜÔ∏è High Clearance Recommended
                      </span>
                    )}
                    {selectedSegment.minimum_vehicle_class && (
                      <span className="px-2 py-1 bg-purple-500/20 text-purple-400 rounded text-sm">
                        üöó Min: {selectedSegment.minimum_vehicle_class}
                      </span>
                    )}
                  </div>
                </div>

                {/* Hazards */}
                {selectedSegment.hazards && selectedSegment.hazards.length > 0 && (
                  <div>
                    <p className="text-gray-400 text-sm font-medium mb-2">‚ö†Ô∏è Hazards</p>
                    <div className="flex flex-wrap gap-2">
                      {selectedSegment.hazards.map((hazard, i) => (
                        <span key={i} className="px-2 py-1 bg-red-500/20 text-red-400 rounded text-sm capitalize">
                          {hazard.replace(/_/g, ' ')}
                        </span>
                      ))}
                    </div>
                  </div>
                )}

                {/* Vehicle Suitability */}
                {vehicle && (
                  <div className={`p-3 rounded-lg ${
                    vehicle.rough_gravel_suitable || selectedSegment.route_type !== 'logging_road'
                      ? 'bg-green-500/20'
                      : 'bg-red-500/20'
                  }`}>
                    <p className={`font-medium ${
                      vehicle.rough_gravel_suitable || selectedSegment.route_type !== 'logging_road'
                        ? 'text-green-400'
                        : 'text-red-400'
                    }`}>
                      {vehicle.rough_gravel_suitable || selectedSegment.route_type !== 'logging_road'
                        ? '‚úÖ Your vehicle is suitable for this route'
                        : '‚ùå Your vehicle may not be suitable for this route'
                      }
                    </p>
                    <p className="text-gray-400 text-sm mt-1">
                      {vehicle.year} {vehicle.make} {vehicle.model}
                    </p>
                  </div>
                )}

                {/* Alternatives */}
                {alternatives.length > 0 && (
                  <div>
                    <p className="text-gray-400 text-sm font-medium mb-2">üîÑ Alternatives (if route unavailable)</p>
                    <div className="space-y-2">
                      {alternatives.map(alt => (
                        <div key={alt.id} className="bg-gray-700/50 rounded-lg p-3">
                          <div className="flex items-center justify-between">
                            <p className="text-white font-medium">
                              {alt.provider_name || alt.alternative_description}
                            </p>
                            <span className="text-gray-400 text-sm">
                              +{alt.additional_time_minutes}min, +${alt.additional_cost_estimate || 0}
                            </span>
                          </div>
                          <p className="text-gray-500 text-sm mt-1">
                            Use when: {alt.trigger_conditions?.join(', ')}
                          </p>
                        </div>
                      ))}
                    </div>
                  </div>
                )}
              </div>
            )}
          </div>
        </div>
      )}

      {/* Transport Providers */}
      <div className="bg-gray-800 rounded-xl p-4">
        <h3 className="text-white font-semibold mb-4">üöå Transport Providers</h3>
        <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-3">
          {providers.map(provider => (
            <div key={provider.id} className="bg-gray-700/50 rounded-lg p-3">
              <div className="flex items-center gap-2 mb-2">
                <span className="text-xl">
                  {provider.provider_type === 'ferry' && '‚õ¥Ô∏è'}
                  {provider.provider_type === 'float_plane' && 'üõ©Ô∏è'}
                  {provider.provider_type === 'water_taxi' && 'üö§'}
                  {provider.provider_type === 'bus' && 'üöå'}
                </span>
                <p className="text-white font-medium">{provider.name}</p>
              </div>
              <p className="text-gray-400 text-sm mb-2">{provider.base_location}</p>
              <div className="flex flex-wrap gap-1">
                {provider.has_live_api && (
                  <span className="px-1.5 py-0.5 bg-green-500/20 text-green-400 rounded text-xs">Live Data</span>
                )}
                {provider.accepts_vehicles && (
                  <span className="px-1.5 py-0.5 bg-blue-500/20 text-blue-400 rounded text-xs">Vehicles OK</span>
                )}
                {provider.accepts_kayaks && (
                  <span className="px-1.5 py-0.5 bg-purple-500/20 text-purple-400 rounded text-xs">Kayaks OK</span>
                )}
              </div>
              {provider.website && (
                <a 
                  href={provider.website} 
                  target="_blank" 
                  rel="noopener noreferrer"
                  className="text-blue-400 text-sm hover:underline mt-2 block"
                >
                  Visit Website ‚Üí
                </a>
              )}
            </div>
          ))}
        </div>
      </div>
    </div>
  );
}

export default RouteExplorer;
