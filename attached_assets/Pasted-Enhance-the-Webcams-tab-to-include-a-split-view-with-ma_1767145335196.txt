// Add to state
const [showMap, setShowMap] = useState(true);
const [hoveredWebcamId, setHoveredWebcamId] = useState<number | null>(null);
const mapContainer = useRef<HTMLDivElement>(null);
const map = useRef<mapboxgl.Map | null>(null);

// Initialize map
useEffect(() => {
  if (!mapContainer.current || !showMap) return;
  if (map.current) return;

  map.current = new mapboxgl.Map({
    container: mapContainer.current,
    style: 'mapbox://styles/mapbox/dark-v11',
    center: [-123.1207, 49.2827],
    zoom: 5,
  });

  map.current.addControl(new mapboxgl.NavigationControl(), 'top-right');

  map.current.on('load', () => {
    updateMapData();
  });

  return () => {
    map.current?.remove();
    map.current = null;
  };
}, [showMap]);

// Update map when filtered webcams change
useEffect(() => {
  if (map.current?.loaded()) {
    updateMapData();
  }
}, [filteredWebcams, hoveredWebcamId]);

function updateMapData() {
  if (!map.current) return;

  // Create GeoJSON from webcams
  const geojson: GeoJSON.FeatureCollection = {
    type: 'FeatureCollection',
    features: filteredWebcams
      .filter(w => w.latitude && w.longitude)
      .map(webcam => ({
        type: 'Feature',
        properties: {
          id: webcam.id,
          name: webcam.name,
          description: webcam.metadata?.view_description,
          source: webcam._category?.source || 'Other',
          isHovered: webcam.id === hoveredWebcamId,
        },
        geometry: {
          type: 'Point',
          coordinates: [webcam.longitude, webcam.latitude],
        },
      })),
  };

  // Remove existing source/layers
  if (map.current.getLayer('webcam-points')) map.current.removeLayer('webcam-points');
  if (map.current.getLayer('webcam-labels')) map.current.removeLayer('webcam-labels');
  if (map.current.getSource('webcams')) map.current.removeSource('webcams');

  // Add source
  map.current.addSource('webcams', {
    type: 'geojson',
    data: geojson,
  });

  // Add points layer
  map.current.addLayer({
    id: 'webcam-points',
    type: 'circle',
    source: 'webcams',
    paint: {
      'circle-radius': [
        'case',
        ['get', 'isHovered'], 12,
        8
      ],
      'circle-color': [
        'match',
        ['get', 'source'],
        'BC Ferries', '#3b82f6',
        'Ski Resorts', '#22c55e',
        'Mountain Passes', '#f97316',
        'Border Crossings', '#a855f7',
        'Bridges & Tunnels', '#eab308',
        'DriveBC Highways', '#ef4444',
        '#6b7280'
      ],
      'circle-stroke-width': 2,
      'circle-stroke-color': '#ffffff',
    },
  });

  // Click handler
  map.current.on('click', 'webcam-points', (e) => {
    if (e.features?.[0]) {
      const id = e.features[0].properties?.id;
      const webcam = filteredWebcams.find(w => w.id === id);
      if (webcam) setSelectedWebcam(webcam);
    }
  });

  // Hover handlers
  map.current.on('mouseenter', 'webcam-points', (e) => {
    map.current!.getCanvas().style.cursor = 'pointer';
    if (e.features?.[0]) {
      const id = e.features[0].properties?.id;
      setHoveredWebcamId(id);
    }
  });

  map.current.on('mouseleave', 'webcam-points', () => {
    map.current!.getCanvas().style.cursor = '';
    setHoveredWebcamId(null);
  });
}

function flyToWebcam(webcam: Webcam) {
  if (map.current && webcam.latitude && webcam.longitude) {
    map.current.flyTo({
      center: [webcam.longitude, webcam.latitude],
      zoom: 14,
      duration: 1000,
    });
  }
}
