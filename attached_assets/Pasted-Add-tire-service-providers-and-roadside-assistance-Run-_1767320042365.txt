-- Verify tire and towing
SELECT 
    sp.name as property,
    ssp.provider_type,
    ssp.business_name,
    ssp.available_24hr as "24hr",
    ssp.overall_rating
FROM staging_service_providers ssp
JOIN staging_properties sp ON sp.id = ssp.property_id
WHERE ssp.provider_type IN ('tire_service', 'towing_heavy')
ORDER BY ssp.provider_type, sp.name;
