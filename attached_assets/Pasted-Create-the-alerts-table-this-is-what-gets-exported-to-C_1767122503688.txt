Create the alerts table - this is what gets exported to CivOS as signals.

Run this SQL:

CREATE TABLE alerts (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    entity_id UUID REFERENCES entities(id) ON DELETE CASCADE,
    region_id VARCHAR(50) REFERENCES geo_regions(id),
    jurisdiction_refs TEXT[],
    alert_type VARCHAR(50) CHECK (alert_type IN (
        'delay', 'cancellation', 'closure', 'capacity', 'weather',
        'emergency', 'advisory', 'maintenance', 'price_change', 
        'new_listing', 'outage', 'restoration', 'hazard', 'evacuation'
    )),
    severity severity_level NOT NULL,
    signal_type VARCHAR(50),
    signal_status signal_status DEFAULT 'active',
    title VARCHAR(255) NOT NULL,
    summary VARCHAR(500),
    message TEXT,
    details JSONB,
    latitude DECIMAL(10, 7),
    longitude DECIMAL(10, 7),
    affected_area JSONB,
    correlation_keys TEXT[],
    effective_from TIMESTAMPTZ DEFAULT NOW(),
    effective_until TIMESTAMPTZ,
    is_active BOOLEAN DEFAULT true,
    acknowledged_at TIMESTAMPTZ,
    resolved_at TIMESTAMPTZ,
    superseded_by UUID REFERENCES alerts(id),
    source_id UUID REFERENCES data_sources(id),
    source_key VARCHAR(100),
    source_url TEXT,
    raw_refs TEXT[],
    confidence DECIMAL(3, 2) DEFAULT 1.00,
    retrieval_method VARCHAR(30),
    parser_version VARCHAR(20),
    raw_checksum VARCHAR(64),
    observed_at TIMESTAMPTZ DEFAULT NOW(),
    reported_at TIMESTAMPTZ DEFAULT NOW(),
    created_at TIMESTAMPTZ DEFAULT NOW(),
    updated_at TIMESTAMPTZ DEFAULT NOW(),
    export_cursor BIGINT GENERATED ALWAYS AS IDENTITY,
    dedupe_key VARCHAR(128)
);

CREATE INDEX idx_alerts_active ON alerts(is_active, severity, created_at DESC);
CREATE INDEX idx_alerts_entity ON alerts(entity_id, created_at DESC);
CREATE INDEX idx_alerts_region ON alerts(region_id, created_at DESC);
CREATE INDEX idx_alerts_cursor ON alerts(export_cursor);
CREATE INDEX idx_alerts_signal_type ON alerts(signal_type);
CREATE INDEX idx_alerts_correlation ON alerts USING GIN(correlation_keys);
CREATE INDEX idx_alerts_jurisdiction ON alerts USING GIN(jurisdiction_refs);
