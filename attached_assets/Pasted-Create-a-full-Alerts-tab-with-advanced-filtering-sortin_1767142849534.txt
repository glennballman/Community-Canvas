import React, { useEffect, useState, useMemo } from 'react';

interface Alert {
  id: number;
  alert_type: string;
  severity: string;
  headline: string;
  description: string;
  region_id: string;
  region_name: string;
  source: string;
  created_at: string;
  updated_at: string;
  expires_at: string | null;
  status: string;
  metadata: Record<string, any>;
}

interface AlertsTabProps {
  regionId?: string;
}

export function AlertsTab({ regionId }: AlertsTabProps) {
  const [alerts, setAlerts] = useState<Alert[]>([]);
  const [loading, setLoading] = useState(true);
  const [selectedAlert, setSelectedAlert] = useState<Alert | null>(null);
  
  // Filters
  const [searchQuery, setSearchQuery] = useState('');
  const [selectedSeverity, setSelectedSeverity] = useState<string>('all');
  const [selectedType, setSelectedType] = useState<string>('all');
  const [selectedRegion, setSelectedRegion] = useState<string>('all');
  const [sortBy, setSortBy] = useState<'newest' | 'oldest' | 'severity'>('newest');
  
  // View mode
  const [viewMode, setViewMode] = useState<'list' | 'compact' | 'timeline'>('list');

  useEffect(() => {
    fetchAlerts();
    // Auto-refresh every 30 seconds
    const interval = setInterval(fetchAlerts, 30000);
    return () => clearInterval(interval);
  }, [regionId]);

  async function fetchAlerts() {
    setLoading(true);
    try {
      const url = `/api/v1/alerts/active?limit=500`;
      const response = await fetch(url);
      const data = await response.json();
      setAlerts(Array.isArray(data) ? data : []);
    } catch (error) {
      console.error('Failed to fetch alerts:', error);
    } finally {
      setLoading(false);
    }
  }

  // Get filter options from data
  const { severities, types, regions } = useMemo(() => {
    const sevSet = new Set<string>();
    const typeSet = new Set<string>();
    const regionSet = new Set<string>();
    
    alerts.forEach(alert => {
      if (alert.severity) sevSet.add(alert.severity);
      if (alert.alert_type) typeSet.add(alert.alert_type);
      if (alert.region_name) regionSet.add(alert.region_name);
    });
    
    return {
      severities: Array.from(sevSet).sort(),
      types: Array.from(typeSet).sort(),
      regions: Array.from(regionSet).sort()
    };
  }, [alerts]);

  // Filter and sort alerts
  const filteredAlerts = useMemo(() => {
    let filtered = alerts;

    // Search
    if (searchQuery) {
      const query = searchQuery.toLowerCase();
      filtered = filtered.filter(a =>
        a.headline?.toLowerCase().includes(query) ||
        a.description?.toLowerCase().includes(query) ||
        a.region_name?.toLowerCase().includes(query)
      );
    }

    // Severity filter
    if (selectedSeverity !== 'all') {
      filtered = filtered.filter(a => a.severity === selectedSeverity);
    }

    // Type filter
    if (selectedType !== 'all') {
      filtered = filtered.filter(a => a.alert_type === selectedType);
    }

    // Region filter
    if (selectedRegion !== 'all') {
      filtered = filtered.filter(a => a.region_name === selectedRegion);
    }

    // Sort
    filtered = [...filtered].sort((a, b) => {
      if (sortBy === 'newest') {
        return new Date(b.created_at).getTime() - new Date(a.created_at).getTime();
      }
      if (sortBy === 'oldest') {
        return new Date(a.created_at).getTime() - new Date(b.created_at).getTime();
      }
      if (sortBy === 'severity') {
        const order = { critical: 0, major: 1, moderate: 2, minor: 3, info: 4 };
        return (order[a.severity as keyof typeof order] || 5) - (order[b.severity as keyof typeof order] || 5);
      }
      return 0;
    });

    return filtered;
  }, [alerts, searchQuery, selectedSeverity, selectedType, selectedRegion, sortBy]);

  // Count by severity for stats
  const severityCounts = useMemo(() => {
    const counts: Record<string, number> = { critical: 0, major: 0, moderate: 0, minor: 0, info: 0 };
    alerts.forEach(a => {
      if (counts[a.severity] !== undefined) {
        counts[a.severity]++;
      }
    });
    return counts;
  }, [alerts]);

  const severityConfig: Record<string, { icon: string; color: string; bg: string; border: string }> = {
    critical: { icon: 'üî¥', color: 'text-red-400', bg: 'bg-red-500/20', border: 'border-red-500' },
    major: { icon: 'üü†', color: 'text-orange-400', bg: 'bg-orange-500/20', border: 'border-orange-500' },
    moderate: { icon: 'üü°', color: 'text-yellow-400', bg: 'bg-yellow-500/20', border: 'border-yellow-500' },
    minor: { icon: 'üîµ', color: 'text-blue-400', bg: 'bg-blue-500/20', border: 'border-blue-500' },
    info: { icon: '‚ÑπÔ∏è', color: 'text-gray-400', bg: 'bg-gray-500/20', border: 'border-gray-500' },
  };

  const typeIcons: Record<string, string> = {
    road_event: 'üõ£Ô∏è',
    incident: 'üö®',
    construction: 'üöß',
    closure: 'üö´',
    weather_warning: '‚õàÔ∏è',
    ferry_delay: '‚õ¥Ô∏è',
    power_outage: '‚ö°',
    earthquake: 'üåã',
    wildfire: 'üî•',
  };

  if (loading) {
    return (
      <div className="space-y-4">
        <div className="bg-gray-800 rounded-xl p-4 animate-pulse">
          <div className="h-10 bg-gray-700 rounded w-64 mb-4"></div>
          <div className="space-y-3">
            {Array(5).fill(0).map((_, i) => (
              <div key={i} className="h-24 bg-gray-700 rounded-lg"></div>
            ))}
          </div>
        </div>
      </div>
    );
  }

  return (
    <div className="space-y-4">
      {/* Header & Stats */}
      <div className="bg-gray-800 rounded-xl p-4">
        <div className="flex flex-col lg:flex-row lg:items-center justify-between gap-4">
          <div>
            <h2 className="text-xl font-bold text-white flex items-center gap-2">
              üö® Active Alerts
              <span className="text-sm font-normal text-gray-400">
                ({filteredAlerts.length} of {alerts.length})
              </span>
            </h2>
            <p className="text-gray-400 text-sm mt-1">
              Real-time alerts across British Columbia
            </p>
          </div>

          {/* Severity Stats */}
          <div className="flex gap-2">
            {Object.entries(severityCounts).map(([sev, count]) => {
              if (count === 0) return null;
              const config = severityConfig[sev];
              return (
                <button
                  key={sev}
                  onClick={() => setSelectedSeverity(selectedSeverity === sev ? 'all' : sev)}
                  className={`px-3 py-1.5 rounded-lg text-sm font-medium transition-colors ${
                    selectedSeverity === sev 
                      ? `${config.bg} ${config.color} ring-2 ring-offset-2 ring-offset-gray-800 ${config.border.replace('border', 'ring')}`
                      : `${config.bg} ${config.color} hover:opacity-80`
                  }`}
                >
                  {config.icon} {count}
                </button>
              );
            })}
          </div>
        </div>

        {/* Search */}
        <div className="mt-4">
          <input
            type="text"
            placeholder="Search alerts by headline, description, or region..."
            value={searchQuery}
            onChange={(e) => setSearchQuery(e.target.value)}
            className="w-full bg-gray-700 text-white rounded-lg px-4 py-2 text-sm border border-gray-600 focus:border-blue-500 focus:outline-none"
          />
        </div>

        {/* Filters Row */}
        <div className="flex flex-wrap gap-3 mt-4">
          {/* Type Filter */}
          <select
            value={selectedType}
            onChange={(e) => setSelectedType(e.target.value)}
            className="bg-gray-700 text-white rounded-lg px-3 py-2 text-sm border border-gray-600"
          >
            <option value="all">All Types</option>
            {types.map(type => (
              <option key={type} value={type}>
                {typeIcons[type] || 'üìã'} {type.replace(/_/g, ' ')}
              </option>
            ))}
          </select>

          {/* Region Filter */}
          <select
            value={selectedRegion}
            onChange={(e) => setSelectedRegion(e.target.value)}
            className="bg-gray-700 text-white rounded-lg px-3 py-2 text-sm border border-gray-600"
          >
            <option value="all">All Regions</option>
            {regions.map(region => (
              <option key={region} value={region}>{region}</option>
            ))}
          </select>

          {/* Sort */}
          <select
            value={sortBy}
            onChange={(e) => setSortBy(e.target.value as any)}
            className="bg-gray-700 text-white rounded-lg px-3 py-2 text-sm border border-gray-600"
          >
            <option value="newest">Newest First</option>
            <option value="oldest">Oldest First</option>
            <option value="severity">By Severity</option>
          </select>

          {/* View Mode */}
          <div className="flex bg-gray-700 rounded-lg p-1 ml-auto">
            <button
              onClick={() => setViewMode('list')}
              className={`px-3 py-1 rounded text-sm ${
                viewMode === 'list' ? 'bg-blue-600 text-white' : 'text-gray-400 hover:text-white'
              }`}
            >
              üìã List
            </button>
            <button
              onClick={() => setViewMode('compact')}
              className={`px-3 py-1 rounded text-sm ${
                viewMode === 'compact' ? 'bg-blue-600 text-white' : 'text-gray-400 hover:text-white'
              }`}
            >
              üìë Compact
            </button>
            <button
              onClick={() => setViewMode('timeline')}
              className={`px-3 py-1 rounded text-sm ${
                viewMode === 'timeline' ? 'bg-blue-600 text-white' : 'text-gray-400 hover:text-white'
              }`}
            >
              üìÖ Timeline
            </button>
          </div>
        </div>

        {/* Clear Filters */}
        {(searchQuery || selectedSeverity !== 'all' || selectedType !== 'all' || selectedRegion !== 'all') && (
          <button
            onClick={() => {
              setSearchQuery('');
              setSelectedSeverity('all');
              setSelectedType('all');
              setSelectedRegion('all');
            }}
            className="mt-3 text-sm text-blue-400 hover:text-blue-300"
          >
            ‚úï Clear all filters
          </button>
        )}
      </div>

      {/* Alert List */}
      {viewMode === 'list' && (
        <div className="space-y-3">
          {filteredAlerts.map(alert => (
            <AlertCard
              key={alert.id}
              alert={alert}
              config={severityConfig}
              typeIcons={typeIcons}
              onClick={() => setSelectedAlert(alert)}
            />
          ))}
        </div>
      )}

      {/* Compact View */}
      {viewMode === 'compact' && (
        <div className="bg-gray-800 rounded-xl overflow-hidden">
          <table className="w-full">
            <thead>
              <tr className="border-b border-gray-700 text-left text-gray-400 text-sm">
                <th className="px-4 py-3">Severity</th>
                <th className="px-4 py-3">Type</th>
                <th className="px-4 py-3">Headline</th>
                <th className="px-4 py-3">Region</th>
                <th className="px-4 py-3">Time</th>
              </tr>
            </thead>
            <tbody className="divide-y divide-gray-700">
              {filteredAlerts.map(alert => {
                const config = severityConfig[alert.severity] || severityConfig.info;
                return (
                  <tr 
                    key={alert.id} 
                    className="hover:bg-gray-700/50 cursor-pointer"
                    onClick={() => setSelectedAlert(alert)}
                  >
                    <td className="px-4 py-3">
                      <span className={`px-2 py-1 rounded text-xs font-medium ${config.bg} ${config.color}`}>
                        {config.icon} {alert.severity}
                      </span>
                    </td>
                    <td className="px-4 py-3 text-sm text-gray-300">
                      {typeIcons[alert.alert_type] || 'üìã'} {alert.alert_type?.replace(/_/g, ' ')}
                    </td>
                    <td className="px-4 py-3 text-sm text-white truncate max-w-md">
                      {alert.headline}
                    </td>
                    <td className="px-4 py-3 text-sm text-gray-400">
                      {alert.region_name}
                    </td>
                    <td className="px-4 py-3 text-sm text-gray-500">
                      {getTimeAgo(new Date(alert.created_at))}
                    </td>
                  </tr>
                );
              })}
            </tbody>
          </table>
        </div>
      )}

      {/* Timeline View */}
      {viewMode === 'timeline' && (
        <div className="relative">
          <div className="absolute left-8 top-0 bottom-0 w-0.5 bg-gray-700"></div>
          <div className="space-y-4">
            {filteredAlerts.map((alert, idx) => {
              const config = severityConfig[alert.severity] || severityConfig.info;
              const showDate = idx === 0 || 
                new Date(alert.created_at).toDateString() !== 
                new Date(filteredAlerts[idx - 1].created_at).toDateString();
              
              return (
                <div key={alert.id}>
                  {showDate && (
                    <div className="ml-16 mb-2 text-sm text-gray-400 font-medium">
                      {new Date(alert.created_at).toLocaleDateString('en-US', {
                        weekday: 'long',
                        month: 'long',
                        day: 'numeric'
                      })}
                    </div>
                  )}
                  <div className="flex items-start gap-4">
                    <div className={`w-4 h-4 rounded-full ${config.bg} ${config.border} border-2 mt-1 z-10`}></div>
                    <div className="text-xs text-gray-500 w-16 pt-1">
                      {new Date(alert.created_at).toLocaleTimeString('en-US', {
                        hour: 'numeric',
                        minute: '2-digit'
                      })}
                    </div>
                    <div 
                      className={`flex-1 bg-gray-800 rounded-lg p-4 border-l-4 ${config.border} cursor-pointer hover:bg-gray-700/50`}
                      onClick={() => setSelectedAlert(alert)}
                    >
                      <div className="flex items-center gap-2 mb-1">
                        <span className={`text-xs font-semibold uppercase ${config.color}`}>
                          {alert.severity}
                        </span>
                        <span className="text-gray-600">‚Ä¢</span>
                        <span className="text-xs text-gray-400">
                          {typeIcons[alert.alert_type]} {alert.alert_type?.replace(/_/g, ' ')}
                        </span>
                      </div>
                      <h4 className="text-white font-medium">{alert.headline}</h4>
                      <p className="text-gray-400 text-sm mt-1">{alert.region_name}</p>
                    </div>
                  </div>
                </div>
              );
            })}
          </div>
        </div>
      )}

      {/* Empty State */}
      {filteredAlerts.length === 0 && (
        <div className="bg-gray-800 rounded-xl p-12 text-center">
          <span className="text-4xl">‚úÖ</span>
          <p className="text-gray-300 mt-2">No alerts match your filters</p>
          <button
            onClick={() => {
              setSearchQuery('');
              setSelectedSeverity('all');
              setSelectedType('all');
              setSelectedRegion('all');
            }}
            className="mt-4 text-blue-400 hover:text-blue-300"
          >
            Clear filters
          </button>
        </div>
      )}

      {/* Detail Modal */}
      {selectedAlert && (
        <AlertDetailModal
          alert={selectedAlert}
          config={severityConfig}
          typeIcons={typeIcons}
          onClose={() => setSelectedAlert(null)}
        />
      )}
    </div>
  );
}

interface AlertCardProps {
  alert: Alert;
  config: Record<string, { icon: string; color: string; bg: string; border: string }>;
  typeIcons: Record<string, string>;
  onClick: () => void;
}

function AlertCard({ alert, config, typeIcons, onClick }: AlertCardProps) {
  const sevConfig = config[alert.severity] || config.info;
  
  return (
    <div
      className={`bg-gray-800 rounded-xl p-4 border-l-4 ${sevConfig.border} cursor-pointer hover:bg-gray-700/50 transition-colors`}
      onClick={onClick}
    >
      <div className="flex items-start justify-between gap-4">
        <div className="flex-1">
          <div className="flex items-center gap-2 flex-wrap mb-2">
            <span className={`px-2 py-1 rounded text-xs font-semibold uppercase ${sevConfig.bg} ${sevConfig.color}`}>
              {sevConfig.icon} {alert.severity}
            </span>
            <span className="text-gray-500">‚Ä¢</span>
            <span className="text-sm text-gray-400">
              {typeIcons[alert.alert_type] || 'üìã'} {alert.alert_type?.replace(/_/g, ' ')}
            </span>
            <span className="text-gray-500">‚Ä¢</span>
            <span className="text-sm text-gray-400">üìç {alert.region_name}</span>
          </div>
          
          <h3 className="text-white font-medium text-lg">{alert.headline}</h3>
          
          {alert.description && (
            <p className="text-gray-400 text-sm mt-2 line-clamp-2">{alert.description}</p>
          )}
          
          <div className="flex items-center gap-4 mt-3 text-xs text-gray-500">
            <span>üïê {getTimeAgo(new Date(alert.created_at))}</span>
            {alert.source && <span>üì° {alert.source}</span>}
          </div>
        </div>
        
        <span className="text-gray-500 text-lg">‚Üí</span>
      </div>
    </div>
  );
}

interface AlertDetailModalProps {
  alert: Alert;
  config: Record<string, { icon: string; color: string; bg: string; border: string }>;
  typeIcons: Record<string, string>;
  onClose: () => void;
}

function AlertDetailModal({ alert, config, typeIcons, onClose }: AlertDetailModalProps) {
  const sevConfig = config[alert.severity] || config.info;
  
  useEffect(() => {
    function handleKeyDown(e: KeyboardEvent) {
      if (e.key === 'Escape') onClose();
    }
    window.addEventListener('keydown', handleKeyDown);
    return () => window.removeEventListener('keydown', handleKeyDown);
  }, [onClose]);
  
  return (
    <div className="fixed inset-0 bg-black/80 z-50 flex items-center justify-center p-4" onClick={onClose}>
      <div 
        className="max-w-2xl w-full bg-gray-900 rounded-xl overflow-hidden"
        onClick={e => e.stopPropagation()}
      >
        {/* Header */}
        <div className={`px-6 py-4 ${sevConfig.bg} border-b border-gray-700`}>
          <div className="flex items-center justify-between">
            <div className="flex items-center gap-3">
              <span className={`px-3 py-1 rounded text-sm font-semibold uppercase ${sevConfig.color} bg-gray-900/50`}>
                {sevConfig.icon} {alert.severity}
              </span>
              <span className="text-gray-300">
                {typeIcons[alert.alert_type] || 'üìã'} {alert.alert_type?.replace(/_/g, ' ')}
              </span>
            </div>
            <button onClick={onClose} className="text-gray-400 hover:text-white text-2xl">
              ‚úï
            </button>
          </div>
        </div>
        
        {/* Content */}
        <div className="p-6">
          <h2 className="text-xl font-bold text-white mb-4">{alert.headline}</h2>
          
          {alert.description && (
            <div className="mb-6">
              <h4 className="text-gray-400 text-sm uppercase mb-2">Description</h4>
              <p className="text-gray-300">{alert.description}</p>
            </div>
          )}
          
          <div className="grid grid-cols-2 gap-4">
            <div>
              <h4 className="text-gray-400 text-sm uppercase mb-1">Region</h4>
              <p className="text-white">üìç {alert.region_name}</p>
            </div>
            <div>
              <h4 className="text-gray-400 text-sm uppercase mb-1">Source</h4>
              <p className="text-white">üì° {alert.source || 'Unknown'}</p>
            </div>
            <div>
              <h4 className="text-gray-400 text-sm uppercase mb-1">Created</h4>
              <p className="text-white">üïê {new Date(alert.created_at).toLocaleString()}</p>
            </div>
            {alert.expires_at && (
              <div>
                <h4 className="text-gray-400 text-sm uppercase mb-1">Expires</h4>
                <p className="text-white">‚è∞ {new Date(alert.expires_at).toLocaleString()}</p>
              </div>
            )}
          </div>
          
          {alert.metadata && Object.keys(alert.metadata).length > 0 && (
            <div className="mt-6 p-4 bg-gray-800 rounded-lg">
              <h4 className="text-gray-400 text-sm uppercase mb-2">Additional Details</h4>
              <pre className="text-xs text-gray-300 overflow-x-auto">
                {JSON.stringify(alert.metadata, null, 2)}
              </pre>
            </div>
          )}
        </div>
      </div>
    </div>
  );
}

function getTimeAgo(date: Date): string {
  const seconds = Math.floor((new Date().getTime() - date.getTime()) / 1000);
  if (seconds < 60) return 'Just now';
  if (seconds < 3600) return `${Math.floor(seconds / 60)}m ago`;
  if (seconds < 86400) return `${Math.floor(seconds / 3600)}h ago`;
  return `${Math.floor(seconds / 86400)}d ago`;
}

export default AlertsTab;
