3. TENANT SWITCHER COMPONENT

The tenant switcher dropdown in the sidebar should show:
- Current tenant name and type
- Dropdown to switch between user's tenants
- Visual indicator of tenant type (community vs business)
```tsx
function TenantSwitcher({ current, memberships, onSwitch, collapsed }) {
  const [open, setOpen] = useState(false);
  
  if (!current) return null;
  
  const typeIcons = {
    community: 'ğŸ”ï¸',
    government: 'ğŸ›ï¸', 
    business: 'ğŸ¢',
    individual: 'ğŸ‘¤'
  };

  return (
    <div className="relative">
      <button
        onClick={() => setOpen(!open)}
        className="w-full p-2 bg-white/5 rounded-lg text-left hover:bg-white/10"
      >
        <div className="flex items-center gap-2">
          <span>{typeIcons[current.type] || 'ğŸ“'}</span>
          {!collapsed && (
            <>
              <div className="flex-1 min-w-0">
                <div className="font-medium truncate">{current.name}</div>
                <div className="text-xs text-gray-500 capitalize">{current.type}</div>
              </div>
              <ChevronDown className="w-4 h-4 text-gray-400" />
            </>
          )}
        </div>
      </button>
      
      {open && !collapsed && (
        <div className="absolute top-full left-0 right-0 mt-1 bg-[#1a2744] border border-white/10 rounded-lg shadow-xl z-50">
          {memberships.map((m) => (
            <button
              key={m.tenant_id}
              onClick={() => { onSwitch(m.tenant_id); setOpen(false); }}
              className="w-full p-3 text-left hover:bg-white/10 first:rounded-t-lg last:rounded-b-lg"
            >
              <div className="flex items-center gap-2">
                <span>{typeIcons[m.tenant_type]}</span>
                <span className="truncate">{m.tenant_name}</span>
              </div>
            </button>
          ))}
        </div>
      )}
    </div>
  );
}
```
