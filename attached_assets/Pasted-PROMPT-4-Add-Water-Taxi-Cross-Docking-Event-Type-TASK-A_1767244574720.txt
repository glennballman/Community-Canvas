5. Create the UI component for water crossing display:
```tsx
   {event.waterCrossing && (
     <div className="px-4 pb-4">
       <div className="p-3 bg-blue-500/10 rounded-lg space-y-3">
         {/* Operators */}
         <div>
           <p className="text-sm font-medium mb-2">Contact to Book:</p>
           {event.waterCrossing.operators.map((op, i) => (
             <div key={i} className="flex items-center justify-between text-sm mb-1">
               <span>{op.name}</span>
               <a href={`tel:${op.phone}`} className="text-primary flex items-center gap-1">
                 <Phone className="h-3 w-3" />
                 {op.phone}
               </a>
             </div>
           ))}
         </div>
         
         {/* Operating Hours */}
         {event.waterCrossing.operatingHours && (
           <div className="text-sm">
             <span className="text-muted-foreground">Hours: </span>
             {event.waterCrossing.operatingHours.start} - {event.waterCrossing.operatingHours.end}
           </div>
         )}
         
         {/* Restrictions */}
         {event.waterCrossing.restrictions.length > 0 && (
           <div className="text-sm">
             <p className="text-warning font-medium mb-1">Restrictions:</p>
             <ul className="list-disc list-inside text-muted-foreground">
               {event.waterCrossing.restrictions.map((r, i) => (
                 <li key={i}>{r}</li>
               ))}
             </ul>
           </div>
         )}
         
         {/* Advance Booking Warning */}
         {event.waterCrossing.bookingRequired && event.waterCrossing.advanceBookingDays && (
           <div className="p-2 bg-warning/20 rounded text-sm text-warning">
             ⚠️ Book at least {event.waterCrossing.advanceBookingDays} days in advance
           </div>
         )}
       </div>
     </div>
   )}
```
