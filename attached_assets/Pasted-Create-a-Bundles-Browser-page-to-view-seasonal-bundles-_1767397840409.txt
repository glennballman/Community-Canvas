Create a Bundles Browser page to view seasonal bundles, their services, pricing, and seasonality.

## Create file: client/src/pages/services/BundlesBrowser.tsx
```tsx
import React, { useState, useEffect } from 'react';

interface BundleItem {
  serviceId: string;
  serviceName: string;
  serviceSlug: string;
  serviceDescription: string;
  quantity: number;
  sortOrder: number;
}

interface BundlePricing {
  basePrice: number;
  discountFactor: number;
  mobilizationSurcharge: number;
  remoteMultiplier: number;
  notes: string;
}

interface BundleSeasonality {
  climateRegionId: string;
  climateRegionName: string;
  earliestWeek: number;
  latestWeek: number;
  hardStop: boolean;
  notes: string;
}

interface Bundle {
  id: string;
  name: string;
  slug: string;
  description: string;
  context: string;
  isSubscription: boolean;
  billingPeriod: string;
  itemCount: number;
  pricing: BundlePricing | null;
}

interface BundleDetail extends Bundle {
  items: BundleItem[];
  seasonality: BundleSeasonality[];
}

interface Community {
  id: string;
  name: string;
  region: string;
  climateRegionName: string;
  remoteMultiplier: number;
}

interface RunType {
  id: string;
  slug: string;
  name: string;
  description: string;
  climateRegionName: string;
  earliestWeek: number;
  latestWeek: number;
  hardStop: boolean;
  isEmergency: boolean;
  priorityWeight: number;
  serviceCount: number;
}

interface Stats {
  bundles: number;
  communities: number;
  currentWeek: number;
}

export default function BundlesBrowser() {
  const [bundles, setBundles] = useState<Bundle[]>([]);
  const [selectedBundle, setSelectedBundle] = useState<BundleDetail | null>(null);
  const [communities, setCommunities] = useState<Community[]>([]);
  const [runTypes, setRunTypes] = useState<RunType[]>([]);
  const [selectedCommunity, setSelectedCommunity] = useState<string>('');
  const [stats, setStats] = useState<Stats | null>(null);
  const [loading, setLoading] = useState(true);
  const [detailLoading, setDetailLoading] = useState(false);
  const [viewTab, setViewTab] = useState<'bundles' | 'calendar'>('bundles');

  // Load initial data
  useEffect(() => {
    async function loadData() {
      setLoading(true);
      try {
        const [statsRes, bundlesRes, communitiesRes, runTypesRes] = await Promise.all([
          fetch('/api/service-runs/stats'),
          fetch('/api/service-runs/bundles'),
          fetch('/api/service-runs/communities'),
          fetch('/api/service-runs/run-types')
        ]);
        
        const [statsData, bundlesData, communitiesData, runTypesData] = await Promise.all([
          statsRes.json(),
          bundlesRes.json(),
          communitiesRes.json(),
          runTypesRes.json()
        ]);
        
        if (statsData.success) setStats(statsData.stats);
        if (bundlesData.success) setBundles(bundlesData.bundles);
        if (communitiesData.success) setCommunities(communitiesData.communities);
        if (runTypesData.success) setRunTypes(runTypesData.runTypes);
      } catch (err) {
        console.error('Failed to load bundles:', err);
      } finally {
        setLoading(false);
      }
    }
    loadData();
  }, []);

  // Load bundle details
  async function loadBundleDetail(slug: string) {
    setDetailLoading(true);
    try {
      const res = await fetch(`/api/service-runs/bundles/${slug}`);
      const data = await res.json();
      
      if (data.success) {
        setSelectedBundle(data.bundle);
      }
    } catch (err) {
      console.error('Failed to load bundle detail:', err);
    } finally {
      setDetailLoading(false);
    }
  }

  // Format discount percentage
  function formatDiscount(factor: number): string {
    const discount = Math.round((1 - factor) * 100);
    return `${discount}% off`;
  }

  // Check if bundle is in season
  function isInSeason(seasonality: BundleSeasonality[], currentWeek: number): boolean {
    if (!seasonality || seasonality.length === 0) return true;
    
    return seasonality.some(s => {
      if (s.earliestWeek <= s.latestWeek) {
        return currentWeek >= s.earliestWeek && currentWeek <= s.latestWeek;
      } else {
        // Wraps around year end
        return currentWeek >= s.earliestWeek || currentWeek <= s.latestWeek;
      }
    });
  }

  // Get season status badge
  function getSeasonBadge(seasonality: BundleSeasonality[], currentWeek: number): React.ReactNode {
    if (!seasonality || seasonality.length === 0) {
      return <span className="text-xs bg-gray-600 text-gray-300 px-2 py-0.5 rounded">Year-round</span>;
    }
    
    const inSeason = isInSeason(seasonality, currentWeek);
    const s = seasonality[0];
    
    if (inSeason) {
      return (
        <span className="text-xs bg-green-500/20 text-green-400 px-2 py-0.5 rounded border border-green-500/50">
          ‚úì In Season (Weeks {s.earliestWeek}-{s.latestWeek})
        </span>
      );
    } else {
      return (
        <span className="text-xs bg-gray-600 text-gray-400 px-2 py-0.5 rounded">
          Weeks {s.earliestWeek}-{s.latestWeek}
          {s.hardStop && ' (Hard Stop)'}
        </span>
      );
    }
  }

  // Week to month approximation
  function weekToMonth(week: number): string {
    const months = ['Jan', 'Feb', 'Mar', 'Apr', 'May', 'Jun', 'Jul', 'Aug', 'Sep', 'Oct', 'Nov', 'Dec'];
    const monthIndex = Math.floor((week - 1) / 4.33);
    return months[Math.min(11, Math.max(0, monthIndex))];
  }

  if (loading) {
    return (
      <div className="p-6 text-center text-gray-400">
        Loading bundles...
      </div>
    );
  }

  return (
    <div className="p-6">
      {/* Header */}
      <div className="flex items-center justify-between mb-6">
        <div>
          <h1 className="text-2xl font-bold text-white">Service Bundles</h1>
          <p className="text-gray-400">Pre-configured service packages for seasonal property care</p>
        </div>
        <div className="flex items-center gap-4">
          <div className="bg-gray-800 rounded-lg px-4 py-2 text-center">
            <div className="text-lg font-bold text-blue-400">Week {stats?.currentWeek}</div>
            <div className="text-xs text-gray-500">Current</div>
          </div>
        </div>
      </div>

      {/* Stats Bar */}
      <div className="grid grid-cols-4 gap-4 mb-6">
        <div className="bg-gray-800 rounded-lg p-4 text-center">
          <div className="text-2xl font-bold text-white">{bundles.length}</div>
          <div className="text-sm text-gray-400">Bundles</div>
        </div>
        <div className="bg-gray-800 rounded-lg p-4 text-center">
          <div className="text-2xl font-bold text-white">{bundles.filter(b => b.isSubscription).length}</div>
          <div className="text-sm text-gray-400">Subscriptions</div>
        </div>
        <div className="bg-gray-800 rounded-lg p-4 text-center">
          <div className="text-2xl font-bold text-white">{communities.length}</div>
          <div className="text-sm text-gray-400">Communities</div>
        </div>
        <div className="bg-gray-800 rounded-lg p-4 text-center">
          <div className="text-2xl font-bold text-white">{runTypes.length}</div>
          <div className="text-sm text-gray-400">Run Types</div>
        </div>
      </div>

      {/* Tabs */}
      <div className="flex gap-2 mb-6">
        <button
          onClick={() => setViewTab('bundles')}
          className={`px-4 py-2 rounded-lg font-medium transition-colors ${
            viewTab === 'bundles' 
              ? 'bg-blue-600 text-white' 
              : 'bg-gray-800 text-gray-400 hover:bg-gray-700'
          }`}
        >
          üì¶ Bundles
        </button>
        <button
          onClick={() => setViewTab('calendar')}
          className={`px-4 py-2 rounded-lg font-medium transition-colors ${
            viewTab === 'calendar' 
              ? 'bg-blue-600 text-white' 
              : 'bg-gray-800 text-gray-400 hover:bg-gray-700'
          }`}
        >
          üìÖ Seasonal Calendar
        </button>
      </div>

      {/* Bundles Tab */}
      {viewTab === 'bundles' && (
        <div className="grid grid-cols-1 lg:grid-cols-3 gap-6">
          {/* Bundles List */}
          <div className="lg:col-span-2 space-y-4">
            {/* Filter */}
            <div className="bg-gray-800 rounded-lg p-4 flex items-center gap-4">
              <select
                value={selectedCommunity}
                onChange={(e) => setSelectedCommunity(e.target.value)}
                className="bg-gray-700 border border-gray-600 rounded-lg px-4 py-2 text-white"
              >
                <option value="">All Communities</option>
                {communities.map(c => (
                  <option key={c.id} value={c.id}>{c.name} ({c.region})</option>
                ))}
              </select>
              <span className="text-gray-500 text-sm">
                Showing {bundles.length} bundles
              </span>
            </div>

            {/* Bundle Cards */}
            {bundles.map(bundle => (
              <div
                key={bundle.id}
                onClick={() => loadBundleDetail(bundle.slug)}
                className={`bg-gray-800 rounded-lg p-4 cursor-pointer hover:bg-gray-750 transition-colors border ${
                  selectedBundle?.slug === bundle.slug 
                    ? 'border-blue-500' 
                    : 'border-gray-700 hover:border-gray-600'
                }`}
              >
                <div className="flex items-start justify-between mb-2">
                  <div className="flex-1">
                    <div className="flex items-center gap-2 mb-1">
                      <h3 className="font-semibold text-white">{bundle.name}</h3>
                      {bundle.isSubscription && (
                        <span className="text-xs bg-purple-500/20 text-purple-400 px-2 py-0.5 rounded border border-purple-500/50">
                          üîÑ Subscription
                        </span>
                      )}
                    </div>
                    <p className="text-sm text-gray-400 line-clamp-2">{bundle.description}</p>
                  </div>
                </div>

                <div className="flex flex-wrap items-center gap-3 mt-3">
                  <span className="text-sm bg-gray-700 px-2 py-1 rounded text-gray-300">
                    üìã {bundle.itemCount} services
                  </span>
                  
                  {bundle.pricing && (
                    <>
                      <span className="text-sm bg-green-500/20 px-2 py-1 rounded text-green-400">
                        üí∞ {formatDiscount(bundle.pricing.discountFactor)}
                      </span>
                      <span className="text-sm bg-gray-700 px-2 py-1 rounded text-gray-300">
                        üöõ ${bundle.pricing.mobilizationSurcharge}
                      </span>
                      {bundle.pricing.remoteMultiplier > 1 && (
                        <span className="text-sm bg-yellow-500/20 px-2 py-1 rounded text-yellow-400">
                          üìç {bundle.pricing.remoteMultiplier}x remote
                        </span>
                      )}
                    </>
                  )}
                  
                  <span className="text-xs bg-blue-500/20 text-blue-400 px-2 py-1 rounded capitalize">
                    {bundle.context}
                  </span>
                </div>
              </div>
            ))}
          </div>

          {/* Bundle Detail Panel */}
          <div className="lg:col-span-1">
            {selectedBundle ? (
              <div className="bg-gray-800 rounded-lg p-4 sticky top-4">
                {detailLoading ? (
                  <div className="text-center py-8 text-gray-500">Loading...</div>
                ) : (
                  <>
                    {/* Header */}
                    <div className="pb-4 border-b border-gray-700 mb-4">
                      <div className="flex items-center gap-2 mb-2">
                        <span className="text-2xl">üì¶</span>
                        <h3 className="text-lg font-bold text-white">{selectedBundle.name}</h3>
                      </div>
                      <p className="text-gray-400 text-sm">{selectedBundle.description}</p>
                      
                      <div className="flex flex-wrap gap-2 mt-3">
                        {selectedBundle.isSubscription && (
                          <span className="text-xs bg-purple-500/20 text-purple-400 px-2 py-1 rounded border border-purple-500/50">
                            üîÑ {selectedBundle.billingPeriod} Subscription
                          </span>
                        )}
                        {selectedBundle.seasonality && selectedBundle.seasonality.length > 0 && (
                          getSeasonBadge(selectedBundle.seasonality, stats?.currentWeek || 1)
                        )}
                      </div>
                    </div>

                    {/* Pricing Summary */}
                    {selectedBundle.pricing && (
                      <div className="mb-4">
                        <h4 className="text-sm font-medium text-gray-300 mb-2">üí∞ Pricing</h4>
                        <div className="bg-gray-700 rounded p-3">
                          {selectedBundle.pricing.basePrice > 0 ? (
                            <div className="text-2xl font-bold text-green-400 mb-1">
                              ${selectedBundle.pricing.basePrice.toLocaleString()}
                              {selectedBundle.isSubscription && (
                                <span className="text-sm font-normal text-gray-400">/{selectedBundle.billingPeriod}</span>
                              )}
                            </div>
                          ) : (
                            <div className="text-lg font-bold text-green-400 mb-1">
                              {formatDiscount(selectedBundle.pricing.discountFactor)} bundle discount
                            </div>
                          )}
                          <div className="text-sm text-gray-400 space-y-1">
                            <div className="flex justify-between">
                              <span>Mobilization:</span>
                              <span className="text-white">${selectedBundle.pricing.mobilizationSurcharge}</span>
                            </div>
                            {selectedBundle.pricing.remoteMultiplier > 1 && (
                              <div className="flex justify-between">
                                <span>Remote Multiplier:</span>
                                <span className="text-white">{selectedBundle.pricing.remoteMultiplier}x</span>
                              </div>
                            )}
                          </div>
                          {selectedBundle.pricing.notes && (
                            <div className="text-xs text-gray-500 mt-2 italic">
                              {selectedBundle.pricing.notes}
                            </div>
                          )}
                        </div>
                      </div>
                    )}

                    {/* Seasonality */}
                    {selectedBundle.seasonality && selectedBundle.seasonality.length > 0 && (
                      <div className="mb-4">
                        <h4 className="text-sm font-medium text-gray-300 mb-2">üìÖ Seasonality</h4>
                        <div className="space-y-2">
                          {selectedBundle.seasonality.map((s, i) => (
                            <div key={i} className="bg-gray-700 rounded p-2 text-sm">
                              <div className="flex items-center justify-between mb-1">
                                <span className="text-white">{s.climateRegionName}</span>
                                <span className="text-gray-400">
                                  {weekToMonth(s.earliestWeek)} - {weekToMonth(s.latestWeek)}
                                </span>
                              </div>
                              <div className="text-xs text-gray-500">
                                Weeks {s.earliestWeek}-{s.latestWeek}
                                {s.hardStop && (
                                  <span className="text-red-400 ml-2">‚ö†Ô∏è Hard Stop</span>
                                )}
                              </div>
                              {s.notes && (
                                <div className="text-xs text-gray-500 mt-1 italic">{s.notes}</div>
                              )}
                            </div>
                          ))}
                        </div>
                      </div>
                    )}

                    {/* Included Services */}
                    <div className="mb-4">
                      <h4 className="text-sm font-medium text-gray-300 mb-2">
                        üìã Included Services ({selectedBundle.items.length})
                      </h4>
                      <div className="space-y-2 max-h-64 overflow-y-auto">
                        {selectedBundle.items
                          .sort((a, b) => a.sortOrder - b.sortOrder)
                          .map((item, i) => (
                            <div key={i} className="bg-gray-700 rounded p-2">
                              <div className="flex items-center justify-between">
                                <span className="text-white text-sm">{item.serviceName}</span>
                                {item.quantity > 1 && (
                                  <span className="text-xs bg-blue-500/20 text-blue-400 px-2 py-0.5 rounded">
                                    √ó{item.quantity}
                                  </span>
                                )}
                              </div>
                              <div className="text-xs text-gray-500 mt-1 line-clamp-1">
                                {item.serviceDescription}
                              </div>
                            </div>
                          ))}
                      </div>
                    </div>

                    {/* Actions */}
                    <div className="pt-4 border-t border-gray-700 space-y-2">
                      <button className="w-full bg-blue-600 hover:bg-blue-700 text-white py-2 rounded text-sm">
                        Create Service Run with Bundle
                      </button>
                      <button className="w-full bg-gray-700 hover:bg-gray-600 text-white py-2 rounded text-sm">
                        Get Quote for My Property
                      </button>
                    </div>
                  </>
                )}
              </div>
            ) : (
              <div className="bg-gray-800 rounded-lg p-8 text-center text-gray-500 sticky top-4">
                <div className="text-4xl mb-3">üì¶</div>
                <p>Select a bundle to view details</p>
              </div>
            )}
          </div>
        </div>
      )}

      {/* Calendar Tab */}
      {viewTab === 'calendar' && (
        <div className="space-y-6">
          {/* Seasonal Calendar Visualization */}
          <div className="bg-gray-800 rounded-lg p-4">
            <h3 className="text-lg font-semibold text-white mb-4">Coastal Wet Region - Seasonal Calendar</h3>
            
            {/* Month headers */}
            <div className="flex mb-2">
              <div className="w-48 shrink-0" />
              {['Jan', 'Feb', 'Mar', 'Apr', 'May', 'Jun', 'Jul', 'Aug', 'Sep', 'Oct', 'Nov', 'Dec'].map(month => (
                <div key={month} className="flex-1 text-center text-xs text-gray-500">{month}</div>
              ))}
            </div>

            {/* Current week indicator */}
            <div className="flex mb-4 relative">
              <div className="w-48 shrink-0 text-xs text-gray-500">Current Week ({stats?.currentWeek})</div>
              <div className="flex-1 relative h-2 bg-gray-700 rounded">
                <div 
                  className="absolute top-0 h-full w-1 bg-blue-500 rounded"
                  style={{ left: `${((stats?.currentWeek || 1) / 52) * 100}%` }}
                />
              </div>
            </div>

            {/* Run Types as Gantt-style bars */}
            <div className="space-y-3">
              {runTypes.map(rt => {
                const startPercent = ((rt.earliestWeek - 1) / 52) * 100;
                let widthPercent: number;
                if (rt.earliestWeek <= rt.latestWeek) {
                  widthPercent = ((rt.latestWeek - rt.earliestWeek + 1) / 52) * 100;
                } else {
                  // Wraps around year
                  widthPercent = ((52 - rt.earliestWeek + rt.latestWeek + 1) / 52) * 100;
                }
                
                const colors = rt.isEmergency 
                  ? 'bg-red-500/40 border-red-500' 
                  : rt.hardStop 
                    ? 'bg-yellow-500/40 border-yellow-500'
                    : 'bg-green-500/40 border-green-500';
                
                return (
                  <div key={rt.id} className="flex items-center">
                    <div className="w-48 shrink-0 pr-4">
                      <div className="text-sm text-white truncate">{rt.name.replace(' Run', '')}</div>
                      <div className="text-xs text-gray-500">
                        {rt.isEmergency && 'üö® Emergency'}
                        {rt.hardStop && !rt.isEmergency && '‚ö†Ô∏è Hard Stop'}
                        {!rt.isEmergency && !rt.hardStop && `${rt.serviceCount} services`}
                      </div>
                    </div>
                    <div className="flex-1 relative h-6 bg-gray-700 rounded">
                      <div 
                        className={`absolute top-0 h-full rounded border ${colors}`}
                        style={{ 
                          left: `${startPercent}%`, 
                          width: `${Math.min(widthPercent, 100 - startPercent)}%` 
                        }}
                      />
                      {/* Handle wrap-around for storm response */}
                      {rt.earliestWeek > rt.latestWeek && (
                        <div 
                          className={`absolute top-0 h-full rounded border ${colors}`}
                          style={{ 
                            left: 0, 
                            width: `${(rt.latestWeek / 52) * 100}%` 
                          }}
                        />
                      )}
                    </div>
                  </div>
                );
              })}
            </div>

            {/* Legend */}
            <div className="flex gap-6 mt-6 pt-4 border-t border-gray-700">
              <div className="flex items-center gap-2">
                <div className="w-4 h-4 bg-green-500/40 border border-green-500 rounded" />
                <span className="text-sm text-gray-400">Standard Window</span>
              </div>
              <div className="flex items-center gap-2">
                <div className="w-4 h-4 bg-yellow-500/40 border border-yellow-500 rounded" />
                <span className="text-sm text-gray-400">Hard Stop</span>
              </div>
              <div className="flex items-center gap-2">
                <div className="w-4 h-4 bg-red-500/40 border border-red-500 rounded" />
                <span className="text-sm text-gray-400">Emergency</span>
              </div>
              <div className="flex items-center gap-2">
                <div className="w-1 h-4 bg-blue-500 rounded" />
                <span className="text-sm text-gray-400">Current Week</span>
              </div>
            </div>
          </div>

          {/* Run Types Detail */}
          <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
            {runTypes.map(rt => (
              <div key={rt.id} className="bg-gray-800 rounded-lg p-4">
                <div className="flex items-start justify-between mb-2">
                  <h4 className="font-semibold text-white">{rt.name}</h4>
                  {rt.isEmergency && (
                    <span className="text-xs bg-red-500/20 text-red-400 px-2 py-0.5 rounded">
                      üö® Emergency
                    </span>
                  )}
                </div>
                <p className="text-sm text-gray-400 mb-3">{rt.description}</p>
                <div className="flex flex-wrap gap-2 text-xs">
                  <span className="bg-gray-700 px-2 py-1 rounded text-gray-300">
                    üìÖ Weeks {rt.earliestWeek}-{rt.latestWeek}
                  </span>
                  <span className="bg-gray-700 px-2 py-1 rounded text-gray-300">
                    üìã {rt.serviceCount} services
                  </span>
                  <span className="bg-gray-700 px-2 py-1 rounded text-gray-300">
                    üéØ Priority {rt.priorityWeight}
                  </span>
                  {rt.hardStop && (
                    <span className="bg-yellow-500/20 px-2 py-1 rounded text-yellow-400">
                      ‚ö†Ô∏è Hard Stop
                    </span>
                  )}
                </div>
              </div>
            ))}
          </div>
        </div>
      )}
    </div>
  );
}
```
