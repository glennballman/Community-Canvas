2. GLOBAL IMPERSONATION BANNER
Create a component that appears at the TOP of the entire app (above everything) when impersonating:
```tsx
// ImpersonationBanner.tsx
function ImpersonationBanner() {
  const [session, setSession] = useState(null);
  
  useEffect(() => {
    checkImpersonationStatus();
  }, []);
  
  async function checkImpersonationStatus() {
    try {
      const res = await fetch('/api/admin/impersonation/status', { credentials: 'include' });
      if (res.ok) {
        const data = await res.json();
        if (data.is_impersonating) {
          setSession(data);
        }
      }
    } catch (e) {}
  }
  
  async function stopImpersonation() {
    await fetch('/api/admin/impersonation/stop', { method: 'POST', credentials: 'include' });
    window.location.href = '/admin/impersonation';
  }
  
  if (!session) return null;
  
  return (
    <div className="fixed top-0 left-0 right-0 z-[9999] bg-amber-500 text-amber-950">
      <div className="max-w-screen-xl mx-auto px-4 py-2 flex items-center justify-between">
        <div className="flex items-center gap-3">
          <span className="w-2 h-2 bg-amber-900 rounded-full animate-pulse" />
          <span className="font-bold">IMPERSONATING:</span>
          <span>{session.tenant_name}</span>
        </div>
        <div className="flex items-center gap-4">
          <span className="text-sm opacity-75">
            Expires: {new Date(session.expires_at).toLocaleTimeString()}
          </span>
          <button
            onClick={stopImpersonation}
            className="px-3 py-1 bg-amber-900 text-amber-50 rounded text-sm font-medium hover:bg-amber-950"
          >
            Stop Impersonation
          </button>
        </div>
      </div>
    </div>
  );
}
```
