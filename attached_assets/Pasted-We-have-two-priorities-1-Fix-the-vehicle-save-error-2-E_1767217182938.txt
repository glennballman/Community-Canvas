// In RoadTripsTab.tsx or wherever trips are listed

// Add a helper function to check qualification
const [qualifications, setQualifications] = useState<Record<string, boolean>>({});

useEffect(() => {
  if (currentParticipantId) {
    checkAllTripQualifications();
  }
}, [currentParticipantId, trips]);

async function checkAllTripQualifications() {
  const results: Record<string, boolean> = {};
  for (const trip of trips) {
    try {
      const res = await fetch('/api/v1/planning/assess/participant-trip', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ participant_id: currentParticipantId, trip_id: trip.id })
      });
      const data = await res.json();
      results[trip.id] = data.qualified !== false;
    } catch {
      results[trip.id] = true; // Assume qualified if can't check
    }
  }
  setQualifications(results);
}

// In the trip card, add badge:
<div className="absolute top-3 right-3">
  {qualifications[trip.id] ? (
    <span className="px-2 py-1 bg-green-500/20 text-green-400 rounded-full text-xs font-medium">
      ‚úì Qualified
    </span>
  ) : (
    <span className="px-2 py-1 bg-orange-500/20 text-orange-400 rounded-full text-xs font-medium">
      üîí Skills Needed
    </span>
  )}
</div>
