## Step 1: Update Property Card for Mobile

Create file: client/src/components/PropertyCard.tsx
```tsx
import React from 'react';
import { Link } from 'react-router-dom';

interface Property {
    id: number;
    name: string;
    city: string;
    region: string;
    totalSpots: number;
    nightlyRate: number | null;
    rvScore: number;
    crewScore: number;
    truckerScore: number;
    hasMechanic: boolean;
    hasPower: boolean;
    petsAllowed: boolean;
    thumbnailUrl?: string;
}

interface PropertyCardProps {
    property: Property;
    compact?: boolean;
}

export default function PropertyCard({ property, compact = false }: PropertyCardProps) {
    // Determine primary score badge
    const primaryScore = Math.max(property.rvScore, property.crewScore, property.truckerScore);
    const primaryType = property.rvScore === primaryScore ? 'RV' 
        : property.crewScore === primaryScore ? 'Crew' 
        : 'Trucker';
    const scoreColor = primaryType === 'RV' ? 'bg-green-600' 
        : primaryType === 'Crew' ? 'bg-blue-600' 
        : 'bg-orange-600';

    if (compact) {
        // Compact horizontal card for mobile lists
        return (
            <Link to={`/staging/${property.id}`} className="block">
                <div className="bg-gray-800 rounded-lg p-3 flex gap-3 active:bg-gray-700">
                    {/* Thumbnail */}
                    <div className="w-20 h-20 flex-shrink-0 bg-gray-700 rounded-lg overflow-hidden">
                        {property.thumbnailUrl ? (
                            <img 
                                src={property.thumbnailUrl} 
                                alt={property.name}
                                className="w-full h-full object-cover"
                            />
                        ) : (
                            <div className="w-full h-full flex items-center justify-center text-2xl">
                                üèïÔ∏è
                            </div>
                        )}
                    </div>

                    {/* Content */}
                    <div className="flex-1 min-w-0">
                        <div className="flex items-start justify-between gap-2">
                            <h3 className="text-white font-medium truncate">{property.name}</h3>
                            {primaryScore > 0 && (
                                <span className={`${scoreColor} text-white text-xs px-1.5 py-0.5 rounded flex-shrink-0`}>
                                    {primaryScore}
                                </span>
                            )}
                        </div>
                        <p className="text-gray-400 text-sm truncate">{property.city}</p>
                        
                        <div className="flex items-center gap-2 mt-1">
                            {property.hasMechanic && <span className="text-green-400 text-xs">üîß</span>}
                            {property.hasPower && <span className="text-yellow-400 text-xs">‚ö°</span>}
                            {property.petsAllowed && <span className="text-purple-400 text-xs">üêï</span>}
                            <span className="text-gray-500 text-xs">{property.totalSpots} spots</span>
                        </div>

                        {property.nightlyRate && (
                            <p className="text-green-400 font-medium text-sm mt-1">
                                ${property.nightlyRate}/night
                            </p>
                        )}
                    </div>
                </div>
            </Link>
        );
    }

    // Full card for desktop/tablet
    return (
        <Link to={`/staging/${property.id}`} className="block group">
            <div className="bg-gray-800 rounded-lg overflow-hidden hover:ring-2 hover:ring-blue-500 transition-all">
                {/* Image */}
                <div className="aspect-video bg-gray-700 relative">
                    {property.thumbnailUrl ? (
                        <img 
                            src={property.thumbnailUrl} 
                            alt={property.name}
                            className="w-full h-full object-cover"
                        />
                    ) : (
                        <div className="w-full h-full flex items-center justify-center text-4xl">
                            üèïÔ∏è
                        </div>
                    )}

                    {/* Score Badge */}
                    {primaryScore > 0 && (
                        <div className={`absolute top-2 right-2 ${scoreColor} text-white px-2 py-1 rounded-lg text-sm font-medium`}>
                            {primaryType}: {primaryScore}
                        </div>
                    )}

                    {/* Mechanic Badge */}
                    {property.hasMechanic && (
                        <div className="absolute top-2 left-2 bg-green-600 text-white px-2 py-1 rounded-lg text-xs">
                            üîß Mechanic
                        </div>
                    )}
                </div>

                {/* Content */}
                <div className="p-4">
                    <h3 className="text-white font-semibold text-lg group-hover:text-blue-400 transition-colors">
                        {property.name}
                    </h3>
                    <p className="text-gray-400">{property.city}, {property.region}</p>

                    <div className="flex items-center gap-3 mt-3 text-sm">
                        <span className="text-gray-500">{property.totalSpots} spots</span>
                        {property.hasPower && <span className="text-yellow-400">‚ö° Power</span>}
                        {property.petsAllowed && <span className="text-purple-400">üêï Pets</span>}
                    </div>

                    {property.nightlyRate && (
                        <div className="mt-3 flex items-center justify-between">
                            <span className="text-green-400 font-bold text-lg">
                                ${property.nightlyRate}
                                <span className="text-gray-500 font-normal text-sm">/night</span>
                            </span>
                        </div>
                    )}
                </div>
            </div>
        </Link>
    );
}
```

## Step 2: Create Mobile Filter Sheet

Create file: client/src/components/MobileFilterSheet.tsx
```tsx
import React from 'react';

interface MobileFilterSheetProps {
    isOpen: boolean;
    onClose: () => void;
    children: React.ReactNode;
    title?: string;
    onApply?: () => void;
    onClear?: () => void;
}

export default function MobileFilterSheet({ 
    isOpen, 
    onClose, 
    children, 
    title = 'Filters',
    onApply,
    onClear
}: MobileFilterSheetProps) {
    if (!isOpen) return null;

    return (
        <div className="fixed inset-0 z-50 lg:hidden">
            {/* Backdrop */}
            <div 
                className="absolute inset-0 bg-black/50"
                onClick={onClose}
            />

            {/* Sheet */}
            <div className="absolute bottom-0 left-0 right-0 bg-gray-800 rounded-t-2xl max-h-[85vh] overflow-hidden flex flex-col">
                {/* Handle */}
                <div className="flex justify-center py-2">
                    <div className="w-10 h-1 bg-gray-600 rounded-full" />
                </div>

                {/* Header */}
                <div className="flex items-center justify-between px-4 py-2 border-b border-gray-700">
                    <button
                        onClick={onClear}
                        className="text-gray-400 hover:text-white"
                    >
                        Clear
                    </button>
                    <h2 className="text-lg font-semibold text-white">{title}</h2>
                    <button
                        onClick={onClose}
                        className="text-gray-400 hover:text-white"
                    >
                        ‚úï
                    </button>
                </div>

                {/* Content */}
                <div className="flex-1 overflow-y-auto p-4">
                    {children}
                </div>

                {/* Footer */}
                <div className="p-4 border-t border-gray-700">
                    <button
                        onClick={() => { onApply?.(); onClose(); }}
                        className="w-full bg-blue-600 hover:bg-blue-700 text-white py-3 rounded-lg font-medium"
                    >
                        Apply Filters
                    </button>
                </div>
            </div>
        </div>
    );
}
```

This creates:
