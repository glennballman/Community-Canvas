import React, { useEffect, useState } from 'react';

interface WeatherData {
  location: string;
  temperature: number;
  feelsLike: number;
  condition: string;
  humidity: number;
  windSpeed: number;
  windDirection: string;
  observedAt: string;
  forecast: ForecastDay[];
  warnings: string[];
}

interface ForecastDay {
  day: string;
  high: number;
  low: number;
  condition: string;
  pop: number;
}

interface WeatherWidgetProps {
  regionId?: string;
}

    try {
      const response = await fetch(`/api/v1/weather?region=${regionId || 'bc'}`);

  if (lower.includes('snow')) return 'ğŸŒ¨ï¸';
  if (lower.includes('rain') || lower.includes('shower')) return 'ğŸŒ§ï¸';
  if (lower.includes('thunder') || lower.includes('storm')) return 'â›ˆï¸';
  if (lower.includes('overcast') || lower.includes('cloudy')) return 'â˜ï¸';
  if (lower.includes('partly')) return 'â›…';
  if (lower.includes('sun') || lower.includes('clear')) return 'â˜€ï¸';
  if (lower.includes('fog') || lower.includes('mist')) return 'ğŸŒ«ï¸';

}
