```

2. **Find ALL indexes with legacy names:**

```

3. **Find tables that DON'T follow cc_ prefix convention:**
```sql

FROM pg_tables 
WHERE schemaname = 'public' 
  AND tablename NOT LIKE 'cc_%'

```

4. **Find columns with problematic naming patterns:**
```sql
SELECT table_name, column_name
FROM information_schema.columns
WHERE table_schema = 'public'
  AND (column_name LIKE '%booker%'
       OR column_name LIKE '%unified%'
       OR column_name LIKE 'num_%'
       OR (column_name LIKE '%_at' AND column_name NOT IN ('created_at', 'updated_at', 'deleted_at')))
ORDER BY table_name, column_name;
```

5. **Find foreign keys referencing old table names:**
```sql
SELECT 
    tc.constraint_name,
    tc.table_name,
    kcu.column_name,
    ccu.table_name AS foreign_table_name
FROM information_schema.table_constraints AS tc
JOIN information_schema.key_column_usage AS kcu
    ON tc.constraint_name = kcu.constraint_name
JOIN information_schema.constraint_column_usage AS ccu
    ON ccu.constraint_name = tc.constraint_name
WHERE tc.constraint_type = 'FOREIGN KEY'
  AND tc.table_schema = 'public'
ORDER BY tc.table_name;
```
