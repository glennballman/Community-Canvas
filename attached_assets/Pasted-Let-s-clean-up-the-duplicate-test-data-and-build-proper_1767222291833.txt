interface VehicleFormProps {
  vehicleId?: string; // If editing existing
  onSave: (vehicle: any) => void;
  onCancel: () => void;
}

export function VehicleForm({ vehicleId, onSave, onCancel }: VehicleFormProps) {
  const [formData, setFormData] = useState<VehicleFormData>({
    nickname: '',
    fleet_number: '',
    year: '',
    make: '',
    model: '',
    color: '',
    license_plate: '',
    vin: '',
    vehicle_class: 'truck',
    drive_type: '4wd',
    fuel_type: 'gas',
    ground_clearance_inches: '',
    length_feet: '',
    height_feet: '',
    passenger_capacity: '',
    towing_capacity_lbs: '',
    has_hitch: false,
    hitch_class: '',
    hitch_ball_size: '',
    has_brake_controller: false,
    trailer_wiring: '',
    has_gooseneck_hitch: false,
    has_fifth_wheel_hitch: false,
    fleet_status: 'available',
    notes: '',
  });
  
  const [photos, setPhotos] = useState<{ type: string; url: string; file?: File }[]>([]);
  const [loading, setLoading] = useState(false);
  const [saving, setSaving] = useState(false);
  const [activeTab, setActiveTab] = useState<'basic' | 'specs' | 'towing' | 'photos'>('basic');

  useEffect(() => {
    if (vehicleId) {
      loadVehicle();
    }
  }, [vehicleId]);

  async function loadVehicle() {
    setLoading(true);
    try {
      const res = await fetch(`/api/v1/planning/vehicles/${vehicleId}`);
      if (res.ok) {
        const data = await res.json();
        setFormData({
          nickname: data.nickname || '',
          fleet_number: data.fleet_number || '',
          year: data.year || '',
          make: data.make || '',
          model: data.model || '',
          color: data.color || '',
          license_plate: data.license_plate || '',
          vin: data.vin || '',
          vehicle_class: data.vehicle_class || 'truck',
          drive_type: data.drive_type || '4wd',
          fuel_type: data.fuel_type || 'gas',
          ground_clearance_inches: data.ground_clearance_inches || '',
          length_feet: data.length_feet || '',
          height_feet: data.height_feet || '',
          passenger_capacity: data.passenger_capacity || '',
          towing_capacity_lbs: data.towing_capacity_lbs || '',
          has_hitch: data.has_hitch || false,
          hitch_class: data.hitch_class || '',
          hitch_ball_size: data.hitch_ball_size || '',
          has_brake_controller: data.has_brake_controller || false,
          trailer_wiring: data.trailer_wiring || '',
          has_gooseneck_hitch: data.has_gooseneck_hitch || false,
          has_fifth_wheel_hitch: data.has_fifth_wheel_hitch || false,
          fleet_status: data.fleet_status || 'available',
          notes: data.notes || '',
        });
      }
      
      // Load photos
      const photosRes = await fetch(`/api/v1/fleet/vehicles/${vehicleId}/photos`);
      if (photosRes.ok) {
        const photosData = await photosRes.json();
        setPhotos(photosData.photos?.map((p: any) => ({ type: p.photo_type, url: p.photo_url })) || []);
      }
    } catch (error) {
      console.error('Error loading vehicle:', error);
    } finally {
      setLoading(false);
    }
  }

  async function handleSave() {
    setSaving(true);
    try {
      const payload = {
        ...formData,
        year: formData.year || null,
        ground_clearance_inches: formData.ground_clearance_inches || null,
        length_feet: formData.length_feet || null,
        height_feet: formData.height_feet || null,
        passenger_capacity: formData.passenger_capacity || null,
        towing_capacity_lbs: formData.towing_capacity_lbs || null,
      };

      let savedVehicle;
      
      if (vehicleId) {
        // Update existing
        const res = await fetch(`/api/v1/fleet/vehicles/${vehicleId}`, {
          method: 'PATCH',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify(payload),
        });
        savedVehicle = await res.json();
      } else {
        // Create new
        const res = await fetch('/api/v1/planning/vehicles', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify(payload),
        });
        savedVehicle = await res.json();
      }

      // Handle photo uploads if any new photos
      // (For now, we'll use URL input - file upload can be added later)
      
      onSave(savedVehicle);
    } catch (error) {
      console.error('Error saving vehicle:', error);
      alert('Failed to save vehicle');
    } finally {
      setSaving(false);
    }
  }

  function updateField(field: keyof VehicleFormData, value: any) {
    setFormData(prev => ({ ...prev, [field]: value }));
  }

  if (loading) {
    return (
      <div className="bg-gray-800 rounded-xl p-12 text-center">
        <span className="text-4xl animate-pulse">üöó</span>

      </div>
    );
  }

  return (
    <div className="bg-gray-800 rounded-xl overflow-hidden">
      {/* Header */}

          <h2 className="text-xl font-bold text-white">
            {vehicleId ? 'Edit Vehicle' : 'Add New Vehicle'}
          </h2>
          <p className="text-gray-400 text-sm">
            {formData.nickname || formData.fleet_number || 'Enter vehicle details'}
          </p>
        </div>
        <div className="flex gap-2">
          <button
            onClick={onCancel}
            className="px-4 py-2 bg-gray-700 hover:bg-gray-600 text-gray-300 rounded-lg"
          >
            Cancel
          </button>
          <button
            onClick={handleSave}
            disabled={saving || !formData.make || !formData.model}
            className="px-4 py-2 bg-blue-600 hover:bg-blue-700 text-white rounded-lg disabled:opacity-50"
          >
            {saving ? 'Saving...' : 'Save Vehicle'}
          </button>
        </div>
      </div>

      <div className="flex border-b border-gray-700">
        {[
          { id: 'basic', label: 'üöó Basic Info' },
          { id: 'specs', label: 'üìê Specs & Dimensions' },
          { id: 'towing', label: 'üîó Towing & Hitch' },
          { id: 'photos', label: 'üì∑ Photos' },
        ].map(tab => (
          <button
            key={tab.id}
            onClick={() => setActiveTab(tab.id as any)}
            className={`px-6 py-3 font-medium transition ${
              activeTab === tab.id
                ? 'text-blue-400 border-b-2 border-blue-400 bg-gray-700/50'
                : 'text-gray-400 hover:text-gray-300'
            }`}
          >
            {tab.label}
          </button>
        ))}
      </div>

      {/* Content */}
      <div className="p-6">
        {/* Basic Info Tab */}
        {activeTab === 'basic' && (
          <div className="space-y-6">
            {/* Identity */}
            <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
              <div>
                <label className="block text-gray-300 text-sm font-medium mb-2">
                  Nickname <span className="text-gray-500">(how you refer to this vehicle)</span>
                </label>
                <input
                  type="text"
                  value={formData.nickname}
                  onChange={(e) => updateField('nickname', e.target.value)}
                  className="w-full bg-gray-700 text-white rounded-lg px-4 py-2 border border-gray-600 focus:border-blue-500 focus:outline-none"
                  placeholder='e.g., "Big Blue", "The Beast"'
                />
              </div>
              <div>
                <label className="block text-gray-300 text-sm font-medium mb-2">
                  Fleet Number <span className="text-gray-500">(internal ID)</span>

                        <option key={hc.value} value={hc.value}>{hc.label}</option>
                      ))}
                    </select>
                  </div>
                  <div>
                    <label className="block text-gray-300 text-sm font-medium mb-2">Ball Size</label>
                    <select
                      value={formData.hitch_ball_size}

                      {BALL_SIZES.map(bs => (
                        <option key={bs.value} value={bs.value}>{bs.label}</option>
                      ))}
                    </select>
                  </div>
                </div>

                <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
                  <div>
                    <label className="block text-gray-300 text-sm font-medium mb-2">Trailer Wiring</label>
                    <select
                      value={formData.trailer_wiring}

                    >
                      {WIRING_TYPES.map(wt => (
                        <option key={wt.value} value={wt.value}>{wt.label}</option>
                      ))}
                    </select>
                  </div>
                  <div className="flex items-center gap-4">
                    <label className="flex items-center gap-2 text-gray-300">
                      <input
                        type="checkbox"
                        checked={formData.has_brake_controller}
                        onChange={(e) => updateField('has_brake_controller', e.target.checked)}
                        className="w-4 h-4 rounded bg-gray-600 border-gray-500"
                      />
                      Brake Controller
                    </label>
                  </div>
                </div>

                <div className="flex flex-wrap gap-4">
                  <label className="flex items-center gap-2 text-gray-300">
                    <input
                      type="checkbox"
                      checked={formData.has_gooseneck_hitch}
                      onChange={(e) => updateField('has_gooseneck_hitch', e.target.checked)}
                      className="w-4 h-4 rounded bg-gray-600 border-gray-500"
                    />
                    Gooseneck Hitch
                  </label>
                  <label className="flex items-center gap-2 text-gray-300">
                    <input
                      type="checkbox"
                      checked={formData.has_fifth_wheel_hitch}
                      onChange={(e) => updateField('has_fifth_wheel_hitch', e.target.checked)}
                      className="w-4 h-4 rounded bg-gray-600 border-gray-500"
                    />
                    Fifth Wheel Hitch
                  </label>
                </div>
              </>
            )}
          </div>
        )}

        {/* Photos Tab */}
        {activeTab === 'photos' && (
          <div className="space-y-6">
            <div className="bg-blue-500/10 border border-blue-500/20 rounded-lg p-4">
              <h4 className="text-blue-400 font-medium mb-2">üì∑ Vehicle Photos</h4>
              <p className="text-gray-300 text-sm">
                Add photos to help identify this vehicle. Recommended: front, rear, both sides, interior.
              </p>
            </div>

            {/* Photo URL Input (simple version) */}
            <div>
              <label className="block text-gray-300 text-sm font-medium mb-2">Primary Photo URL</label>
              <input
                type="url"
                placeholder="https://example.com/photo.jpg"
                className="w-full bg-gray-700 text-white rounded-lg px-4 py-2 border border-gray-600 focus:border-blue-500 focus:outline-none"
              />
              <p className="text-gray-500 text-xs mt-1">
                Paste a URL to an image, or drag & drop photos here (coming soon)
              </p>
            </div>

            {/* Photo Grid Placeholder */}
            <div className="grid grid-cols-4 gap-4">
              {['Front', 'Rear', 'Driver Side', 'Passenger Side', 'Interior Front', 'Interior Rear', 'Cargo Area', 'Damage/Notes'].map((label, i) => (
                <div key={i} className="aspect-video bg-gray-700 rounded-lg border-2 border-dashed border-gray-600 flex flex-col items-center justify-center text-gray-500 hover:border-gray-500 cursor-pointer">
                  <span className="text-2xl">üì∑</span>
                  <span className="text-xs mt-1">{label}</span>
                </div>
              ))}
            </div>
          </div>
        )}

        {/* Notes (always visible at bottom) */}
        <div className="mt-6 pt-6 border-t border-gray-700">
          <label className="block text-gray-300 text-sm font-medium mb-2">Notes</label>
          <textarea
            value={formData.notes}
            onChange={(e) => updateField('notes', e.target.value)}
            className="w-full bg-gray-700 text-white rounded-lg px-4 py-2 border border-gray-600 focus:border-blue-500 focus:outline-none"
            placeholder="Any additional notes about this vehicle..."
            rows={3}
          />
        </div>
      </div>
    </div>
  );
}

function SuitabilityBadge({ label, sublabel, suitable }: { label: string; sublabel?: string; suitable: boolean }) {
  return (
    <div className={`p-3 rounded-lg text-center ${suitable ? 'bg-green-500/20' : 'bg-red-500/20'}`}>
      <span className={`text-lg ${suitable ? 'text-green-400' : 'text-red-400'}`}>
        {suitable ? '‚úì' : '‚úó'}
      </span>
      <p className={`text-sm font-medium ${suitable ? 'text-green-400' : 'text-red-400'}`}>{label}</p>
      {sublabel && <p className="text-xs text-gray-500">{sublabel}</p>}
    </div>
  );
}
