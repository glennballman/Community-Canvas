import React, { useEffect, useState } from 'react';

interface RoadEvent {
  id: string;
  event_type: string;
  severity: string;
  headline: string;
  description: string;
  roads: string;
  location: string;
  created_at: string;
  estimated_end?: string;
}

interface RoadEventsProps {
  regionId?: string;
  maxEvents?: number;
}

export function RoadEvents({ regionId, maxEvents = 6 }: RoadEventsProps) {
  const [events, setEvents] = useState<RoadEvent[]>([]);
  const [loading, setLoading] = useState(true);
  const [filter, setFilter] = useState<string>('all');

  useEffect(() => {
    fetchEvents();
    // Refresh every 5 minutes
    const interval = setInterval(fetchEvents, 5 * 60 * 1000);
    return () => clearInterval(interval);
  }, [regionId]);

  async function fetchEvents() {
    try {
      // Try to fetch from alerts with road_event type
      const url = `/api/v1/alerts/active?type=road_event&limit=${maxEvents}`;
      const response = await fetch(url);
      const data = await response.json();
      
      if (Array.isArray(data) && data.length > 0) {
        setEvents(data);
      } else {
        // Use mock data as fallback
        setEvents(getMockRoadEvents());
      }
    } catch (error) {
      console.error('Failed to fetch road events:', error);
      setEvents(getMockRoadEvents());
    } finally {
      setLoading(false);
    }
  }

  const eventTypeConfig: Record<string, { icon: string; label: string; color: string }> = {
    incident: { icon: 'üö®', label: 'Incident', color: 'text-red-400' },
    construction: { icon: 'üöß', label: 'Construction', color: 'text-orange-400' },
    closure: { icon: 'üö´', label: 'Closure', color: 'text-red-500' },
    weather: { icon: '‚õàÔ∏è', label: 'Weather', color: 'text-blue-400' },
    special_event: { icon: 'üéâ', label: 'Event', color: 'text-purple-400' },
  };

  // Count by type
  const counts = events.reduce((acc, event) => {
    const type = event.event_type?.toLowerCase() || 'incident';
    acc[type] = (acc[type] || 0) + 1;
    return acc;
  }, {} as Record<string, number>);

  const filteredEvents = filter === 'all'
    ? events
    : events.filter(e => e.event_type?.toLowerCase() === filter);

  if (loading) {
    return (
      <div className="bg-gray-800 rounded-xl p-4">
        <div className="animate-pulse space-y-3">
          <div className="h-6 bg-gray-700 rounded w-32"></div>
          <div className="flex gap-2">
            {[1, 2, 3].map(i => (
              <div key={i} className="h-8 bg-gray-700 rounded-full w-20"></div>
            ))}
          </div>
          {[1, 2, 3].map(i => (
            <div key={i} className="h-16 bg-gray-700 rounded-lg"></div>
          ))}
        </div>
      </div>
    );
  }

  return (
    <div className="bg-gray-800 rounded-xl overflow-hidden">
      <div className="px-4 py-3 border-b border-gray-700">
        <div className="flex items-center justify-between">
          <h3 className="font-semibold text-white flex items-center gap-2">
            <span>üõ£Ô∏è</span> Road Events
            <span className="text-sm font-normal text-gray-400">
              ({events.length})
            </span>
          </h3>
          
            href="https://www.drivebc.ca"
            target="_blank"
            rel="noopener noreferrer"
            className="text-xs text-blue-400 hover:text-blue-300"
          >
            DriveBC ‚Üí
          </a>
        </div>

        {/* Filter Pills */}
        <div className="flex gap-2 mt-3 flex-wrap">
          <FilterPill
            active={filter === 'all'}
            onClick={() => setFilter('all')}
            icon="üìã"
            label="All"
            count={events.length}
          />
          {Object.entries(eventTypeConfig).map(([type, config]) => (
            counts[type] > 0 && (
              <FilterPill
                key={type}
                active={filter === type}
                onClick={() => setFilter(type)}
                icon={config.icon}
                label={config.label}
                count={counts[type]}
              />
            )
          ))}
        </div>
      </div>

      <div className="divide-y divide-gray-700 max-h-80 overflow-y-auto">
        {filteredEvents.length === 0 ? (
          <div className="p-6 text-center">
            <span className="text-3xl">‚úÖ</span>
            <p className="text-gray-300 mt-2">No active road events</p>
          </div>
        ) : (
          filteredEvents.slice(0, maxEvents).map(event => (
            <RoadEventItem key={event.id} event={event} config={eventTypeConfig} />
          ))
        )}
      </div>
    </div>
  );
}

interface FilterPillProps {
  active: boolean;
  onClick: () => void;
  icon: string;
  label: string;
  count: number;
}

function FilterPill({ active, onClick, icon, label, count }: FilterPillProps) {
  return (
    <button
      onClick={onClick}
      className={`flex items-center gap-1.5 px-3 py-1.5 rounded-full text-xs font-medium transition-colors ${
        active
          ? 'bg-blue-500/30 text-blue-300 border border-blue-500/50'
          : 'bg-gray-700 text-gray-300 border border-gray-600 hover:bg-gray-600'
      }`}
    >
      <span>{icon}</span>
      <span>{label}</span>
      <span className={`ml-1 px-1.5 py-0.5 rounded-full text-xs ${
        active ? 'bg-blue-500/30' : 'bg-gray-600'
      }`}>
        {count}
      </span>
    </button>
  );
}

interface RoadEventItemProps {
  event: RoadEvent;
  config: Record<string, { icon: string; label: string; color: string }>;
}

function RoadEventItem({ event, config }: RoadEventItemProps) {
  const [expanded, setExpanded] = useState(false);
  const type = event.event_type?.toLowerCase() || 'incident';
  const typeConfig = config[type] || config.incident;

  const severityColors: Record<string, string> = {
    major: 'border-l-red-500 bg-red-500/5',
    moderate: 'border-l-orange-500 bg-orange-500/5',
    minor: 'border-l-yellow-500 bg-yellow-500/5',
  };

  const severityClass = severityColors[event.severity?.toLowerCase()] || severityColors.minor;

  return (
    <div
      className={`p-3 border-l-4 ${severityClass} cursor-pointer hover:bg-gray-700/30 transition-colors`}
      onClick={() => setExpanded(!expanded)}
    >
      <div className="flex items-start gap-3">
        <span className="text-lg flex-shrink-0">{typeConfig.icon}</span>

        <div className="flex-1 min-w-0">
          <div className="flex items-center gap-2 flex-wrap">
            <span className={`text-xs font-semibold uppercase ${typeConfig.color}`}>
              {typeConfig.label}
            </span>
            {event.roads && (
              <>
                <span className="text-gray-600">‚Ä¢</span>
                <span className="text-xs text-gray-400">{event.roads}</span>
              </>
            )}
          </div>

          <h4 className="text-white text-sm mt-1 line-clamp-2">
            {event.headline}
          </h4>

          {expanded && event.description && (
            <p className="text-gray-400 text-xs mt-2 line-clamp-3">
              {event.description}
            </p>
          )}

          {expanded && event.estimated_end && (
            <p className="text-gray-500 text-xs mt-2">
              üïê Est. end: {new Date(event.estimated_end).toLocaleString()}
            </p>
          )}
        </div>

        <span className="text-gray-500 text-sm flex-shrink-0">
          {expanded ? '‚ñ≤' : '‚ñº'}
        </span>
      </div>
    </div>
  );
}
