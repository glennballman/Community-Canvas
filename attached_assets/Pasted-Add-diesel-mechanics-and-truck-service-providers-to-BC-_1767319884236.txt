-- Verify diesel mechanics
SELECT 
    sp.name as property,
    ssp.business_name,
    ssp.provider_name,
    CASE WHEN ssp.is_resident THEN 'âœ“ RESIDENT' ELSE 'Mobile' END as type,
    ssp.available_24hr as "24hr",
    ssp.hourly_rate,
    ssp.overall_rating
FROM staging_service_providers ssp
JOIN staging_properties sp ON sp.id = ssp.property_id
WHERE ssp.provider_type = 'mechanic_diesel'
ORDER BY ssp.is_resident DESC, ssp.overall_rating DESC;
