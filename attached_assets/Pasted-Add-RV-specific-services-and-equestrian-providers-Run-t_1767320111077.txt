-- Verify all RV and equestrian services
SELECT 
    sp.name as property,
    ssp.provider_type,
    ssp.business_name,
    ssp.provider_name,
    ssp.overall_rating
FROM staging_service_providers ssp
JOIN staging_properties sp ON sp.id = ssp.property_id
WHERE ssp.provider_type IN ('mechanic_rv', 'farrier', 'vet_emergency')
ORDER BY ssp.provider_type;
