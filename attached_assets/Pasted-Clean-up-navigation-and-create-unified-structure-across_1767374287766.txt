## Step 3: Create Quick Links Component

Create client/src/components/QuickLinks.tsx for consistent cross-app navigation:
```tsx
import React from 'react';
import { Link, useLocation } from 'react-router-dom';

interface QuickLinksProps {
  variant?: 'admin' | 'public' | 'host';
}

export default function QuickLinks({ variant = 'public' }: QuickLinksProps) {
  const location = useLocation();
  
  const links = {
    admin: [
      { path: '/', label: 'Public Site', icon: 'ğŸŒ' },
      { path: '/host/dashboard', label: 'Host View', icon: 'ğŸ ' },
      { path: '/admin/civos', label: 'CivOS Ops', icon: 'ğŸ›¡ï¸' },
    ],
    public: [
      { path: '/staging', label: 'Find Staging', icon: 'ğŸ”' },
      { path: '/staging/map', label: 'Map', icon: 'ğŸ—ºï¸' },
      { path: '/host/login', label: 'Host Login', icon: 'ğŸ ' },
    ],
    host: [
      { path: '/', label: 'Public Site', icon: 'ğŸŒ' },
      { path: '/staging', label: 'Find Staging', icon: 'ğŸ”' },
      { path: '/admin', label: 'Admin', icon: 'âš™ï¸' },
    ],
  };

  return (
    <div className="flex gap-2 text-sm">
      {links[variant].map(link => (
        <Link
          key={link.path}
          to={link.path}
          className={`px-3 py-1 rounded ${
            location.pathname === link.path 
              ? 'bg-blue-600 text-white' 
              : 'bg-gray-700 text-gray-300 hover:bg-gray-600'
          }`}
        >
          {link.icon} {link.label}
        </Link>
      ))}
    </div>
  );
}
```

## Step 5: Create Unified Header for Public Pages

Create client/src/components/PublicHeader.tsx:
```tsx
import React, { useState } from 'react';
import { Link, useLocation } from 'react-router-dom';
import { useAuth } from '../contexts/AuthContext';

export default function PublicHeader() {
  const { user, logout } = useAuth();
  const location = useLocation();
  const [menuOpen, setMenuOpen] = useState(false);

  const mainNav = [
    { path: '/staging', label: 'Find Staging' },
    { path: '/staging/map', label: 'Map' },
    { path: '/staging/chamber', label: 'Local Services' },
  ];

  const isActive = (path: string) => location.pathname === path;

  return (
    <header className="bg-gray-800 border-b border-gray-700 sticky top-0 z-40">
      <div className="max-w-7xl mx-auto px-4">
        <div className="flex items-center justify-between h-14">
          {/* Logo */}
          <Link to="/" className="flex items-center gap-2">
            <span className="text-2xl">ğŸš</span>
            <span className="font-bold text-white hidden sm:block">BC Staging Network</span>
          </Link>

          {/* Desktop Nav */}
          <nav className="hidden md:flex items-center gap-1">
            {mainNav.map(item => (
              <Link
                key={item.path}
                to={item.path}
                className={`px-3 py-2 rounded-lg text-sm ${
                  isActive(item.path)
                    ? 'bg-blue-600 text-white'
                    : 'text-gray-300 hover:bg-gray-700'
                }`}
              >
                {item.label}
              </Link>
            ))}
          </nav>

          {/* User Menu */}
          <div className="flex items-center gap-2">
            {user ? (
              <div className="relative">
                <button
                  onClick={() => setMenuOpen(!menuOpen)}
                  className="flex items-center gap-2 px-3 py-2 rounded-lg bg-gray-700 text-white text-sm"
                >
                  <span className="w-6 h-6 bg-blue-600 rounded-full flex items-center justify-center text-xs">
                    {user.firstName?.[0] || user.email[0].toUpperCase()}
                  </span>
                  <span className="hidden sm:block">{user.firstName || 'Account'}</span>
                </button>
                
                {menuOpen && (
                  <div className="absolute right-0 mt-2 w-48 bg-gray-800 border border-gray-700 rounded-lg shadow-xl py-1">
                    <Link to="/account" className="block px-4 py-2 text-gray-300 hover:bg-gray-700">My Account</Link>
                    <Link to="/staging/bookings" className="block px-4 py-2 text-gray-300 hover:bg-gray-700">My Bookings</Link>
                    <Link to="/account/vehicles" className="block px-4 py-2 text-gray-300 hover:bg-gray-700">My Vehicles</Link>
                    {(user.userType === 'host' || user.userType === 'admin') && (
                      <Link to="/host/dashboard" className="block px-4 py-2 text-green-400 hover:bg-gray-700">Host Dashboard</Link>
                    )}
                    {user.userType === 'admin' && (
                      <Link to="/admin" className="block px-4 py-2 text-purple-400 hover:bg-gray-700">Admin Console</Link>
                    )}
                    <hr className="border-gray-700 my-1" />
                    <button onClick={logout} className="block w-full text-left px-4 py-2 text-red-400 hover:bg-gray-700">Sign Out</button>
                  </div>
                )}
              </div>
            ) : (
              <div className="flex items-center gap-2">
                <Link to="/host/login" className="px-3 py-2 text-gray-300 hover:text-white text-sm">
                  Host Login
                </Link>
                <Link to="/host/signup" className="px-3 py-2 bg-blue-600 hover:bg-blue-700 text-white rounded-lg text-sm">
                  List Property
                </Link>
              </div>
            )}
          </div>
        </div>
      </div>
    </header>
  );
}
```
