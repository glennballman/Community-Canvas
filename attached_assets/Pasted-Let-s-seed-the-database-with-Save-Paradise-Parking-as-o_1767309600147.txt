-- Add some individual spots
INSERT INTO staging_spots (property_id, spot_number, spot_name, spot_type, length_ft, width_ft, max_combined_length_ft, is_pull_through, has_power, power_amps, is_waterfront, view_type)
SELECT 
    p.id,
    'A' || n,
    CASE 
        WHEN n <= 3 THEN 'Waterfront ' || n
        ELSE 'Back Row ' || (n - 3)
    END,
    CASE 
        WHEN n <= 4 THEN 'parking_oversized'
        ELSE 'parking_standard'
    END,
    CASE WHEN n <= 4 THEN 80 ELSE 50 END,
    15,
    CASE WHEN n <= 4 THEN 85 ELSE 55 END,
    n <= 6,
    n <= 8,
    CASE WHEN n <= 8 THEN 30 ELSE NULL END,
    n <= 3,
    CASE WHEN n <= 3 THEN 'inlet' ELSE NULL END
FROM staging_properties p, generate_series(1, 10) n
WHERE p.name = 'Save Paradise Parking';

SELECT name, canvas_id, crew_score, rv_score, trucker_score, status
